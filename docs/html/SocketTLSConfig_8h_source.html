273/* ============================================================================
274 * TLS Protocol Versions
275 * ============================================================================
276 *
277 * DEFAULT: Strict TLS 1.3-only enforcement (SOCKET_TLS_MIN_VERSION ==
278 * SOCKET_TLS_MAX_VERSION == TLS1_3_VERSION).
279 *
280 * This is the RECOMMENDED configuration for security. TLS 1.3 provides:
281 * - Mandatory Perfect Forward Secrecy (PFS) with ephemeral keys
282 * - Modern AEAD ciphers only (no CBC, RC4, 3DES)
283 * - Simplified handshake (1-RTT, optional 0-RTT)
284 * - Protection against downgrade attacks
285 * - Removal of legacy/vulnerable features (renegotiation, compression)
286 *
287 * LEGACY COMPATIBILITY (NOT RECOMMENDED):
288 * =======================================
289 * If you MUST support legacy systems that cannot upgrade to TLS 1.3, you can
290 * override these constants BEFORE including this header. This is strongly
291 * discouraged and should only be used as a last resort.
292 *
293 * ## Security Implications of Lowering TLS Version
294 *
295 * | Version | Security Status | Known Vulnerabilities |
296 * |------------|-----------------|------------------------|
297 * | TLS 1.3 | SECURE | None known |
298 * | TLS 1.2 | ACCEPTABLE* | Lucky13, ROBOT, CBC issues |
299 * | TLS 1.1 | DEPRECATED | No PFS by default, BEAST variants |
300 * | TLS 1.0 | DEPRECATED | BEAST, POODLE variants |
301 * | SSL 3.0 | BROKEN | POODLE, padding oracle attacks |
302 * | SSL 2.0 | BROKEN | Completely broken, never use |
303 *
304 * *TLS 1.2 is acceptable only with ECDHE+AEAD cipher suites and proper config.
305 *
306 * ## Legacy Override Example (Compile-Time)
307 *
308 * @code{.c}
309 * // BEFORE including SocketTLSConfig.h, define overrides:
310 * // WARNING: This weakens security! Document your justification.
311 * #define SOCKET_TLS_MIN_VERSION TLS1_2_VERSION
312 * #define SOCKET_TLS_MAX_VERSION TLS1_3_VERSION
313 * #include "tls/SocketTLSConfig.h"
314 *
315 * // This allows TLS 1.2 fallback while preferring TLS 1.3
316 * @endcode
317 *
318 * ## Runtime Override (Preferred for Flexibility)
319 *
320 * @code{.c}
321 * // Use SocketTLSConfig_T for runtime configuration:
322 * SocketTLSConfig_T config;
323 * SocketTLS_config_defaults(&config);
324 *
325 * // Override for legacy compatibility (logs security warning):
326 * config.min_version = TLS1_2_VERSION; // Allow TLS 1.2 fallback
327 * // config.max_version remains TLS1_3_VERSION (prefer TLS 1.3)
328 *
329 * SocketTLSContext_T ctx = SocketTLSContext_new(&config);
330 * @endcode
331 *
332 * ## Alternative: Per-Context Override (Most Flexible)
333 *
334 * @code{.c}
335 * // Create with defaults, then adjust specific context:
336 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
337 *
338 * // Lower version for specific legacy connection (emits warning log):
339 * SocketTLSContext_set_min_protocol(ctx, TLS1_2_VERSION);
340 * @endcode
341 *
342 * SECURITY RECOMMENDATION:
343 * - Prefer upgrading legacy systems to TLS 1.3
344 * - If TLS 1.2 is required, use ECDHE key exchange only
345 * - Monitor connections and plan migration timelines
346 * - Audit cipher suites: remove CBC modes, use AEAD only
347 * - Consider separate contexts for legacy vs. modern connections
348 */
