69remove_ws_client (SocketWS_T ws)
70{
71 for (int i = 0; i < ws_count; i++)
72 {
73 if (ws_clients[i] == ws)

232static void
233process_websocket_clients (void)
234{
235 for (int i = 0; i < ws_count; i++)
236 {
237 SocketWS_T ws = ws_clients[i];
238
239 /* Check if connection is still open */
240 if (SocketWS_state (ws) == WS_STATE_CLOSED)
241 {
242 printf ("WebSocket client disconnected\n");
243 remove_ws_client (ws); /* Remove from array first */
244 SocketWS_free (&ws); /* Then free the WebSocket */
245 i--; /* Adjust index after removal */
246 continue;
247 }
248
249 /* Try to receive a message (non-blocking would require poll integration)
250 */
251 if (SocketWS_recv_available (ws))
252 {
253 SocketWS_Message msg;
254 int recv_result = SocketWS_recv_message (ws, &msg);
255
256 if (recv_result > 0)
257 {
258 printf ("Received %s message (%zu bytes)\n",
259 msg.type == WS_OPCODE_TEXT ? "text" : "binary", msg.len);
260
261 /* Echo the message back */
262 if (msg.type == WS_OPCODE_TEXT)
263 {
264 SocketWS_send_text (ws, (const char *)msg.data, msg.len);
265 }
266 else
267 {
268 SocketWS_send_binary (ws, msg.data, msg.len);
269 }
270
271 free (msg.data);
272 }
273 else if (recv_result == 0)
274 {
275 /* Connection closed */
276 printf ("WebSocket client closed connection (code: %d)\n",
277 SocketWS_close_code (ws));
278 remove_ws_client (ws); /* Remove from array first */
279 SocketWS_free (&ws); /* Then free the WebSocket */
280 i--;
281 }
282 }
283 }
