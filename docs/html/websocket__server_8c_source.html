227static void
228process_websocket_clients (void)
229{
230 for (int i = 0; i < ws_count; i++)
231 {
232 SocketWS_T ws = ws_clients[i];
233
234 /* Check if connection is still open */
235 if (SocketWS_state (ws) == WS_STATE_CLOSED)
236 {
237 printf ("WebSocket client disconnected\n");
238 SocketWS_free (&ws);
239 remove_ws_client (ws_clients[i]);
240 i--; /* Adjust index after removal */
241 continue;
242 }
243
244 /* Try to receive a message (non-blocking would require poll integration) */
245 if (SocketWS_recv_available (ws))
246 {
247 SocketWS_Message msg;
248 int recv_result = SocketWS_recv_message (ws, &msg);
249
250 if (recv_result > 0)
251 {
252 printf ("Received %s message (%zu bytes)\n",
253 msg.type == WS_OPCODE_TEXT ? "text" : "binary", msg.len);
254
255 /* Echo the message back */
256 if (msg.type == WS_OPCODE_TEXT)
257 {
258 SocketWS_send_text (ws, (const char *)msg.data, msg.len);
259 }
260 else
261 {
262 SocketWS_send_binary (ws, msg.data, msg.len);
263 }
264
265 free (msg.data);
266 }
267 else if (recv_result == 0)
268 {
269 /* Connection closed */
270 printf ("WebSocket client closed connection (code: %d)\n",
271 SocketWS_close_code (ws));
272 SocketWS_free (&ws);
273 remove_ws_client (ws_clients[i]);
274 i--;
275 }
276 }
277 }
