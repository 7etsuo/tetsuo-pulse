695/* ============================================================================
696 * Certificate Pinning (SPKI SHA256)
697 * ============================================================================
698 *
699 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
700 * SPKI pinning survives certificate renewal when the same key is reused,
701 * making it more maintainable than full certificate pinning.
702 *
703 * Usage:
704 * // Create client context
705 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
706 *
707 * // Pin by hex-encoded SHA256 hash
708 * SocketTLSContext_add_pin_hex(ctx,
709 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
710 *
711 * // Or extract from certificate file
712 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
713 *
714 * // Enable strict enforcement (default, fail if no match)
715 * SocketTLSContext_set_pin_enforcement(ctx, 1);
716 *
717 * Generate pin from certificate:
718 * openssl x509 -in cert.pem -pubkey -noout | \
719 * openssl pkey -pubin -outform DER | \
720 * openssl dgst -sha256 -hex
721 *
722 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
723 * before sharing the context across threads. Verification is read-only
724 * and thread-safe.
725 */

902/* ============================================================================

905 *
906 * CT helps detect mis-issued certificates by requiring them to be logged
