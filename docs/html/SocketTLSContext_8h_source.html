726/* ============================================================================
727 * Certificate Pinning (SPKI SHA256)
728 * ============================================================================
729 *
730 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
731 * SPKI pinning survives certificate renewal when the same key is reused,
732 * making it more maintainable than full certificate pinning.
733 *
734 * Usage:
735 * // Create client context
736 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
737 *
738 * // Pin by hex-encoded SHA256 hash
739 * SocketTLSContext_add_pin_hex(ctx,
740 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
741 *
742 * // Or extract from certificate file
743 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
744 *
745 * // Enable strict enforcement (default, fail if no match)
746 * SocketTLSContext_set_pin_enforcement(ctx, 1);
747 *
748 * Generate pin from certificate:
749 * openssl x509 -in cert.pem -pubkey -noout | \
750 * openssl pkey -pubin -outform DER | \
751 * openssl dgst -sha256 -hex
752 *
753 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
754 * before sharing the context across threads. Verification is read-only
755 * and thread-safe.
756 */

933/* ============================================================================
934 * Certificate Transparency (RFC 6962)
935 * ============================================================================
936 *
937 * CT helps detect mis-issued certificates by requiring them to be logged
938 * in publicly auditable CT logs. Clients verify Signed Certificate
939 * Timestamps (SCTs) embedded in certificates, TLS extensions, or OCSP.
940 *
941 * Requires OpenSSL 1.1.0+ with CT support compiled in.
942 *
943 * Usage:
944 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
945 * SocketTLSContext_enable_ct(ctx, CT_VALIDATION_STRICT);
946 * // ... connections will require valid SCTs
947 *
948 * Thread safety: Configuration is NOT thread-safe - perform before sharing.
