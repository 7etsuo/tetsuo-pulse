629/* ============================================================================
630 * Certificate Pinning (SPKI SHA256)
631 * ============================================================================
632 *
633 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
634 * SPKI pinning survives certificate renewal when the same key is reused,
635 * making it more maintainable than full certificate pinning.
636 *
637 * Usage:
638 * // Create client context
639 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
640 *
641 * // Pin by hex-encoded SHA256 hash
642 * SocketTLSContext_add_pin_hex(ctx,
643 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
644 *
645 * // Or extract from certificate file
646 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
647 *
648 * // Enable strict enforcement (default, fail if no match)
649 * SocketTLSContext_set_pin_enforcement(ctx, 1);
650 *
651 * Generate pin from certificate:
652 * openssl x509 -in cert.pem -pubkey -noout | \
653 * openssl pkey -pubin -outform DER | \
654 * openssl dgst -sha256 -hex
655 *
656 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
657 * before sharing the context across threads. Verification is read-only
658 * and thread-safe.
659 */

807/* ============================================================================

810 *
811 * CT helps detect mis-issued certificates by requiring them to be logged
812 * in publicly auditable CT logs. Clients verify Signed Certificate
