866/* ============================================================================
867 * OCSP Must-Staple Support (RFC 7633)
868 * ============================================================================
869 *
870 * OCSP Must-Staple is a certificate extension (id-pe-tlsfeature, OID
871 * 1.3.6.1.5.5.7.1.24) that indicates the certificate MUST be accompanied
872 * by a valid OCSP stapled response. If a certificate has this extension
873 * and no valid OCSP response is provided, the connection should be rejected.
874 *
875 * This provides stronger revocation checking by making OCSP stapling
876 * mandatory rather than optional, preventing downgrade attacks where an
877 * attacker strips the OCSP response.
878 *
879 * Usage:
880 * // Enable auto-detection (check cert for must-staple extension)
881 * SocketTLSContext_set_ocsp_must_staple(ctx, OCSP_MUST_STAPLE_AUTO);
882 *
883 * // Always require OCSP stapling (regardless of cert extension)
884 * SocketTLSContext_set_ocsp_must_staple(ctx, OCSP_MUST_STAPLE_ALWAYS);
885 *
886 * // Disable must-staple checking (default)
887 * SocketTLSContext_set_ocsp_must_staple(ctx, OCSP_MUST_STAPLE_DISABLED);
888 *
889 * Thread safety: Configuration is NOT thread-safe - perform before sharing.
890 */
891

902 OCSP_MUST_STAPLE_DISABLED
903 = 0,
904 OCSP_MUST_STAPLE_AUTO
905 = 1,
906 OCSP_MUST_STAPLE_ALWAYS
907 = 2

1542/* ============================================================================
1543 * Certificate Pinning (SPKI SHA256)
1544 * ============================================================================
1545 *
1546 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
1547 * SPKI pinning survives certificate renewal when the same key is reused,
1548 * making it more maintainable than full certificate pinning.
1549 *
1550 * Usage:
1551 * // Create client context
1552 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
1553 *
1554 * // Pin by hex-encoded SHA256 hash
1555 * SocketTLSContext_add_pin_hex(ctx,
1556 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
1557 *
1558 * // Or extract from certificate file
1559 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
1560 *
1561 * // Enable strict enforcement (default, fail if no match)
1562 * SocketTLSContext_set_pin_enforcement(ctx, 1);
1563 *
1564 * Generate pin from certificate:
1565 * openssl x509 -in cert.pem -pubkey -noout | \
1566 * openssl pkey -pubin -outform DER | \
1567 * openssl dgst -sha256 -hex
1568 *
1569 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
1570 * before sharing the context across threads. Verification is read-only
1571 * and thread-safe.
1572 */
