894/* ============================================================================
895 * Certificate Pinning (SPKI SHA256)
896 * ============================================================================
897 *
898 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
899 * SPKI pinning survives certificate renewal when the same key is reused,
900 * making it more maintainable than full certificate pinning.
901 *
902 * Usage:
903 * // Create client context
904 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
905 *
906 * // Pin by hex-encoded SHA256 hash
907 * SocketTLSContext_add_pin_hex(ctx,
908 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
909 *
910 * // Or extract from certificate file
911 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
912 *
913 * // Enable strict enforcement (default, fail if no match)
914 * SocketTLSContext_set_pin_enforcement(ctx, 1);
915 *
916 * Generate pin from certificate:
917 * openssl x509 -in cert.pem -pubkey -noout | \
918 * openssl pkey -pubin -outform DER | \
919 * openssl dgst -sha256 -hex
920 *
921 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
922 * before sharing the context across threads. Verification is read-only
923 * and thread-safe.
924 */

1137 *
1138 * CT helps detect mis-issued certificates by requiring them to be logged
