615/* ============================================================================
616 * Certificate Pinning (SPKI SHA256)
617 * ============================================================================
618 *
619 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
620 * SPKI pinning survives certificate renewal when the same key is reused,
621 * making it more maintainable than full certificate pinning.
622 *
623 * Usage:
624 * // Create client context
625 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
626 *
627 * // Pin by hex-encoded SHA256 hash
628 * SocketTLSContext_add_pin_hex(ctx,
629 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
630 *
631 * // Or extract from certificate file
632 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
633 *
634 * // Enable strict enforcement (default, fail if no match)
635 * SocketTLSContext_set_pin_enforcement(ctx, 1);
636 *
637 * Generate pin from certificate:
638 * openssl x509 -in cert.pem -pubkey -noout | \
639 * openssl pkey -pubin -outform DER | \
640 * openssl dgst -sha256 -hex
641 *
642 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
643 * before sharing the context across threads. Verification is read-only
644 * and thread-safe.
645 */

793/* ============================================================================

797 * CT helps detect mis-issued certificates by requiring them to be logged
