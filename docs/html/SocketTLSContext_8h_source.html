425/* ============================================================================
426 * Certificate Pinning (SPKI SHA256)
427 * ============================================================================
428 *
429 * OWASP-recommended approach: Pin the Subject Public Key Info (SPKI) hash.
430 * SPKI pinning survives certificate renewal when the same key is reused,
431 * making it more maintainable than full certificate pinning.
432 *
433 * Usage:
434 * // Create client context
435 * SocketTLSContext_T ctx = SocketTLSContext_new_client("ca-bundle.pem");
436 *
437 * // Pin by hex-encoded SHA256 hash
438 * SocketTLSContext_add_pin_hex(ctx,
439 * "b5bb9d8014a0f9b1d61e21e796d78dccdf1352f23cd32812f4850b878ae4944c");
440 *
441 * // Or extract from certificate file
442 * SocketTLSContext_add_pin_from_cert(ctx, "server.crt");
443 *
444 * // Enable strict enforcement (default, fail if no match)
445 * SocketTLSContext_set_pin_enforcement(ctx, 1);
446 *
447 * Generate pin from certificate:
448 * openssl x509 -in cert.pem -pubkey -noout | \
449 * openssl pkey -pubin -outform DER | \
450 * openssl dgst -sha256 -hex
451 *
452 * Thread safety: Pin configuration is NOT thread-safe. Configure all pins
453 * before sharing the context across threads. Verification is read-only
454 * and thread-safe.
455 */
