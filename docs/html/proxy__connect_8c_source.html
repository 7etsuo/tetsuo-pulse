129main (int argc, char **argv)
130{
131 const char *proxy_url;
132 const char *target_host;
133 int target_port;
134 SocketProxy_Config proxy_config;
135 Socket_T sock = NULL;
136 volatile int result = 0;
137
138 /* Parse arguments */
139 if (argc < 4)
140 {
141 print_usage (argv[0]);
142 return 1;
143 }
144
145 proxy_url = argv[1];
146 target_host = argv[2];
147 target_port = atoi (argv[3]);
148
149 if (target_port <= 0 || target_port > 65535)
150 {
151 fprintf (stderr, "Invalid port: %s\n", argv[3]);
152 return 1;
153 }
154
155 /* Handle SIGPIPE */
156 signal (SIGPIPE, SIG_IGN);
157
158 printf ("Proxy Connect Example\n");
159 printf ("=====================\n\n");
160
161 /* Parse proxy URL */
162 SocketProxy_config_defaults (&proxy_config);
163 if (SocketProxy_parse_url (proxy_url, &proxy_config, NULL) < 0)
164 {
165 fprintf (stderr, "Failed to parse proxy URL: %s\n", proxy_url);
166 return 1;
167 }
168
169 /* Print connection info */
170 printf ("Proxy Type: %s\n", proxy_type_name (proxy_config.type));
171 printf ("Proxy Host: %s\n", proxy_config.host);
172 printf ("Proxy Port: %d\n", proxy_config.port);
173 if (proxy_config.username)
174 {
175 printf ("Username: %s\n", proxy_config.username);
176 printf ("Password: %s\n", proxy_config.password ? "****" : "(none)");
177 }
178 printf ("\n");
179 printf ("Target: %s:%d\n", target_host, target_port);
180 printf ("\n");
181
182 TRY
183 {
184 printf ("Connecting through proxy...\n");
185
186 /* Connect through proxy (blocking) */
187 sock = SocketProxy_connect (&proxy_config, target_host, target_port);
188
189 if (!sock)
190 {
191 fprintf (stderr, "Proxy connection returned NULL\n");
192 result = 1;
193 }
194 else
195 {
196 printf ("SUCCESS: Tunnel established!\n\n");
197
198 /* Demonstrate the connection is working */
199 printf ("Connected to %s:%d through proxy\n", target_host,
200 target_port);
201 printf ("Socket fd: %d\n", Socket_fd (sock));
202 printf ("Local address: %s:%d\n", Socket_getlocaladdr (sock),
203 Socket_getlocalport (sock));
204
205 /* If target is HTTPS (443), demonstrate TLS setup */
206 if (target_port == 443)
207 {
208#if SOCKET_HAS_TLS
209 printf ("\nTarget port is 443 - performing TLS handshake...\n");
210
211 /* Create TLS context */
212 SocketTLSContext_T tls_ctx = SocketTLSContext_new_client (NULL);
213
214 /* Enable TLS on socket */
215 SocketTLS_enable (sock, tls_ctx);
216 SocketTLS_set_hostname (sock, target_host);
217
218 /* Perform TLS handshake */
219 while (SocketTLS_handshake (sock) > 0)
220 {
221 /* Handshake in progress */
222 }
223
224 printf ("TLS Version: %s\n", SocketTLS_get_version (sock));
225 printf ("TLS Cipher: %s\n", SocketTLS_get_cipher (sock));
226
227 /* Send a simple HTTP request over TLS */
228 char request[512];
229 snprintf (request, sizeof (request),
230 "GET / HTTP/1.1\r\n"
231 "Host: %s\r\n"
232 "Connection: close\r\n\r\n",
233 target_host);
234
235 printf ("\nSending HTTPS request...\n");
236 SocketTLS_send (sock, request, strlen (request));
237
238 /* Read response */
239 char response[4096];
240 ssize_t n = SocketTLS_recv (sock, response, sizeof (response) - 1);
241 if (n > 0)
242 {
243 response[n] = '\0';
244 /* Print first 500 chars of response */
245 printf ("\nResponse (first 500 chars):\n");
246 printf ("%.500s\n", response);
247 }
248
249 SocketTLS_shutdown (sock);
250 SocketTLSContext_free (&tls_ctx);
251#else
252 printf ("\nTarget port is 443 but TLS support not compiled in.\n");
253 printf ("Compile with -DENABLE_TLS=ON for HTTPS over proxy "
254 "support.\n");
255#endif
256 }
257 else
258 {
259 /* For non-TLS, send a simple HTTP request */
260 char request[512];
261 snprintf (request, sizeof (request),
262 "GET / HTTP/1.1\r\n"
263 "Host: %s\r\n"
264 "Connection: close\r\n\r\n",
265 target_host);
266
267 printf ("\nSending HTTP request...\n");
268 Socket_sendall (sock, request, strlen (request));
269
270 /* Read response */
271 char response[4096];
272 ssize_t n = Socket_recv (sock, response, sizeof (response) - 1);
273 if (n > 0)

275 response[n] = '\0';
276 /* Print first 500 chars of response */
277 printf ("\nResponse (first 500 chars):\n");
278 printf ("%.500s\n", response);
279 }
280 }
281 }
282 }
283 EXCEPT (SocketProxy_Failed)

285 fprintf (stderr, "Proxy connection failed\n");
286 result = 1;

288 EXCEPT (Socket_Failed)

290 fprintf (stderr, "Socket error\n");

312 END_TRY;
