688static inline int
689http1_tokenbuf_append (HTTP1_TokenBuf *buf, Arena_T arena, char c,
690 size_t max_size)
691{
692 if (buf->len >= max_size)
693 return -1;
694
695 if (buf->len >= buf->capacity)
696 {
697 /* Double capacity */
698 size_t new_capacity = buf->capacity * 2;
699 if (new_capacity > max_size)
700 new_capacity = max_size;
701
702 char *new_data = Arena_alloc (arena, new_capacity, __FILE__, __LINE__);
703 if (!new_data)
704 return -1;
705
706 memmove (new_data, buf->data, buf->len);
707 buf->data = new_data;
708 buf->capacity = new_capacity;
709 }
710
711 buf->data[buf->len++] = c;
712 return 0;
713}
714
736static inline char *
737http1_tokenbuf_terminate (HTTP1_TokenBuf *buf, Arena_T arena, size_t max_size)
738{
739 if (buf->len >= buf->capacity)
740 {
741 /* Need space for null terminator */
742 size_t new_capacity = buf->len + 1;
743 if (new_capacity > max_size + 1)
744 return NULL;
745
746 char *new_data = Arena_alloc (arena, new_capacity, __FILE__, __LINE__);
747 if (!new_data)
748 return NULL;
749
750 memmove (new_data, buf->data, buf->len);
751 buf->data = new_data;
752 buf->capacity = new_capacity;
753 }
754
755 buf->data[buf->len] = '\0';
756 return buf->data;
757}
