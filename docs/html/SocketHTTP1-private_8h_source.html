437static inline int
438http1_tokenbuf_append (HTTP1_TokenBuf *buf, Arena_T arena, char c,
439 size_t max_size)
440{
441 if (buf->len >= max_size)
442 return -1;
443
444 if (buf->len >= buf->capacity)
445 {
446 /* Double capacity */
447 size_t new_capacity = buf->capacity * 2;
448 if (new_capacity > max_size)
449 new_capacity = max_size;
450
451 char *new_data = Arena_alloc (arena, new_capacity, __FILE__, __LINE__);
452 if (!new_data)
453 return -1;
454
455 memcpy (new_data, buf->data, buf->len);
456 buf->data = new_data;
457 buf->capacity = new_capacity;
458 }
459
460 buf->data[buf->len++] = c;
461 return 0;
462}

480static inline char *
481http1_tokenbuf_terminate (HTTP1_TokenBuf *buf, Arena_T arena, size_t max_size)
482{
483 if (buf->len >= buf->capacity)
484 {
485 /* Need space for null terminator */
486 size_t new_capacity = buf->len + 1;
487 if (new_capacity > max_size + 1)
488 return NULL;
489
490 char *new_data = Arena_alloc (arena, new_capacity, __FILE__, __LINE__);
491 if (!new_data)
492 return NULL;
493
494 memcpy (new_data, buf->data, buf->len);
495 buf->data = new_data;
496 buf->capacity = new_capacity;
497 }
498
499 buf->data[buf->len] = '\0';
500 return buf->data;
501}
