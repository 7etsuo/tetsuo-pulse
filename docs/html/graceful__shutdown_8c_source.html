74static void
75signal_handler (int signo)
76{
77 int saved_errno = errno;
78
79 g_last_signal = signo;
80
81 /* Write signal number to pipe - write() is async-signal-safe */
82 char byte = (char)signo;
83 ssize_t ret = write (g_signal_pipe[1], &byte, 1);
84 (void)ret; /* Ignore - best effort in signal handler */
85
86 errno = saved_errno;
87}

103static int
104setup_signal_handling (void)
105{
106 /* Create self-pipe */
107 if (pipe (g_signal_pipe) < 0)
108 {
109 perror ("pipe");
110 return -1;
111 }
112
113 /* Make both ends non-blocking to prevent blocking in signal handler */
114 if (fcntl (g_signal_pipe[0], F_SETFL, O_NONBLOCK) < 0
115 || fcntl (g_signal_pipe[1], F_SETFL, O_NONBLOCK) < 0)
116 {
117 perror ("fcntl O_NONBLOCK");
118 close (g_signal_pipe[0]);
119 close (g_signal_pipe[1]);
120 return -1;
121 }
122
123 /* Set close-on-exec to prevent leaking to child processes */
124 fcntl (g_signal_pipe[0], F_SETFD, FD_CLOEXEC);
125 fcntl (g_signal_pipe[1], F_SETFD, FD_CLOEXEC);
126
127 /* Install signal handlers using sigaction (not signal()) */
128 struct sigaction sa;
129 memset (&sa, 0, sizeof (sa));
130 sa.sa_handler = signal_handler;
131 sigemptyset (&sa.sa_mask);
132 sa.sa_flags = SA_RESTART; /* Restart interrupted syscalls */
133
134 if (sigaction (SIGINT, &sa, NULL) < 0)
135 {
136 perror ("sigaction SIGINT");
137 return -1;
138 }
139
140 if (sigaction (SIGTERM, &sa, NULL) < 0)
141 {
142 perror ("sigaction SIGTERM");
143 return -1;
144 }
145
146 if (sigaction (SIGHUP, &sa, NULL) < 0)
147 {
148 perror ("sigaction SIGHUP");
149 return -1;
150 }
151
152 return 0;
153}

192static int
193handle_client_data (Socket_T client, SocketBuf_T inbuf, SocketBuf_T outbuf)
194{
195 char buf[BUFFER_SIZE];
196 ssize_t n;
197
198 (void)inbuf;
199 (void)outbuf;
200
201 TRY
202 {
203 n = Socket_recv (client, buf, sizeof (buf) - 1);
204 if (n > 0)
205 {
206 buf[n] = '\0';
207 printf ("[%s:%d] Received %zd bytes\n", Socket_getpeeraddr (client),
208 Socket_getpeerport (client), n);
209
210 /* Echo back */
211 Socket_sendall (client, buf, n);
212 return 1; /* Keep connection */
213 }
214 }
215 EXCEPT (Socket_Closed)
216 {
217 printf ("[%s:%d] Client disconnected\n", Socket_getpeeraddr (client),
218 Socket_getpeerport (client));
219 }
220 EXCEPT (Socket_Failed)
221 {
222 printf ("[%s:%d] Socket error: %s\n", Socket_getpeeraddr (client),
223 Socket_getpeerport (client), Socket_GetLastError ());
224 }
225 END_TRY;
226
227 return 0; /* Close connection */
228}

236{
237 /* Variables used across TRY/EXCEPT must be volatile to avoid clobbering */

241 SocketPool_T pool = NULL;
242 Arena_T arena = NULL;
243 int running = 1;
244
245 /* Parse command line */
246 if (argc > 1)
247 {
248 port = atoi (argv[1]);
249 if (port <= 0 || port > 65535)
250 {
251 fprintf (stderr, "Invalid port: %s\n", argv[1]);
252 return 1;
253 }
254 }
255
256 /* Setup signal handling infrastructure */
257 if (setup_signal_handling () < 0)
258 {
259 fprintf (stderr, "Failed to setup signal handling\n");
260 return 1;
261 }
262
263 printf ("=== Graceful Shutdown Example ===\n");
264 printf ("Press Ctrl+C or send SIGTERM to initiate graceful shutdown\n\n");
265
266 TRY
267 {
268 /* Create resources */
269 arena = Arena_new ();
270 server = Socket_new (AF_INET, SOCK_STREAM, 0);
271 poll = SocketPoll_new (MAX_CONNECTIONS + 10);
272 pool = SocketPool_new (arena, MAX_CONNECTIONS, BUFFER_SIZE);
273
274 /* Configure and bind server */
275 Socket_setreuseaddr (server);
276 Socket_bind (server, NULL, port);
277 Socket_listen (server, 128);
278 Socket_setnonblocking (server);
279
280 printf ("Server listening on port %d\n", port);
281 printf ("Drain timeout: %d seconds\n\n", DRAIN_TIMEOUT_MS / 1000);
282
283 /* Add server socket to poll */
284 SocketPoll_add (poll, server, POLL_READ, server);
285
286 /* Main event loop
287 *
288 * Note: We use a short poll timeout and check the signal pipe
289 * separately. This is because SocketPoll only supports Socket_T,
290 * not raw file descriptors. For more sophisticated integration,
291 * you could use poll()/select() directly for both.
292 */
293 while (running)
294 {
295 /* Check signal pipe first (non-blocking) */
296 if (check_signal_pipe ())
297 {
298 printf ("\n[SIGNAL] Received signal %d, initiating graceful "
299 "shutdown...\n",
300 (int)g_last_signal);
301 running = 0;
302 break;
303 }
304
305 /* Poll for socket events with short timeout */
306 SocketEvent_T *events;
307 int nevents = SocketPoll_wait (poll, &events, POLL_TIMEOUT_MS);
308
309 for (int i = 0; i < nevents && running; i++)
310 {
311 /* Check for new connections */
312 if (events[i].socket == server)
313 {
314 Socket_T client = Socket_accept (server);
315 if (client)
316 {
317 Socket_setnonblocking (client);
318 Connection_T conn = SocketPool_add (pool, client);
319 if (conn)
320 {
321 SocketPoll_add (poll, client, POLL_READ, conn);
322 printf ("[%s:%d] Client connected (total: %zu)\n",
323 Socket_getpeeraddr (client),
324 Socket_getpeerport (client),
325 SocketPool_count (pool));
326 }
327 else
328 {
329 printf ("Pool full, rejecting connection\n");
330 Socket_free (&client);
331 }
332 }
333 continue;
334 }
335
336 /* Handle client data */
337 Connection_T conn = events[i].data;
338 if (conn)
339 {
340 Socket_T client = Connection_socket (conn);
341 SocketBuf_T inbuf = Connection_inbuf (conn);
342 SocketBuf_T outbuf = Connection_outbuf (conn);
343
344 if (!handle_client_data (client, inbuf, outbuf))
345 {
346 /* Close connection */
347 SocketPoll_del (poll, client);
348 SocketPool_remove (pool, client);
349 Socket_free (&client);
350 printf ("Connections remaining: %zu\n",
351 SocketPool_count (pool));
352 }
353 }
354 }
355 }
356
357 /* =========================================================================
358 * Graceful Shutdown Phase
359 * ========================================================================= */
360
361 printf ("\n=== Graceful Shutdown ===\n");
362 printf ("Active connections: %zu\n", SocketPool_count (pool));
363
364 /* Stop accepting new connections */
365 SocketPoll_del (poll, server);
366 Socket_free (&server);
367 server = NULL;
368 printf ("Stopped accepting new connections\n");
369
370 /* Initiate drain with timeout */
371 if (SocketPool_count (pool) > 0)
372 {
373 printf ("Draining %zu connections (timeout: %ds)...\n",
374 SocketPool_count (pool), DRAIN_TIMEOUT_MS / 1000);
375
376 SocketPool_drain (pool, DRAIN_TIMEOUT_MS);
377
378 /* Continue processing events until drain completes */
379 while (SocketPool_drain_poll (pool) > 0)
380 {
381 SocketEvent_T *events;
382 int timeout = SocketPool_drain_remaining_ms (pool);
383 if (timeout <= 0)
384 timeout = 100;
385
386 int nevents = SocketPoll_wait (poll, &events, timeout);
387
388 for (int i = 0; i < nevents; i++)
389 {
390 Connection_T conn = events[i].data;
391 if (conn)
392 {
393 Socket_T client = Connection_socket (conn);
394 SocketBuf_T inbuf = Connection_inbuf (conn);
395 SocketBuf_T outbuf = Connection_outbuf (conn);
396
397 /* Process remaining data, then close */
398 handle_client_data (client, inbuf, outbuf);
399
400 SocketPoll_del (poll, client);
401 SocketPool_remove (pool, client);
402 Socket_free (&client);
403
404 printf ("Drained connection (remaining: %zu)\n",
405 SocketPool_count (pool));
406 }
407 }
408 }
409
410 size_t remaining = SocketPool_count (pool);
411 if (remaining > 0)
412 {
413 printf ("Drain timeout - force closing %zu connections\n",
414 remaining);
415 SocketPool_drain_force (pool);
416 }
417 else
418 {
419 printf ("All connections drained gracefully\n");
420 }
421 }
422 else
423 {
424 printf ("No connections to drain\n");
425 }
426 }
427 EXCEPT (Socket_Failed)
428 {
429 fprintf (stderr, "Socket error: %s\n", Socket_GetLastError ());
430 }
431 EXCEPT (SocketPoll_Failed)
432 {
433 fprintf (stderr, "Poll error: %s\n", Socket_GetLastError ());
434 }
435 EXCEPT (SocketPool_Failed)
436 {
437 fprintf (stderr, "Pool error: %s\n", Socket_GetLastError ());
438 }
439 END_TRY;
440
441 /* Cleanup */
442 printf ("\n=== Cleanup ===\n");
443
444 if (server)
445 {
446 Socket_free (&server);
447 printf ("Server socket closed\n");
448 }
449
450 if (pool)
451 {
452 SocketPool_free (&pool);
453 printf ("Connection pool freed\n");
454 }
455
456 if (poll)
457 {
458 SocketPoll_free (&poll);
459 printf ("Poll instance freed\n");
460 }
461
462 if (arena)
463 {
464 Arena_dispose (&arena);
465 printf ("Arena disposed\n");
466 }
467
468 cleanup_signal_handling ();
469 printf ("Signal handling cleaned up\n");
470
471 printf ("\nGraceful shutdown complete.\n");
472 return 0;
