78static void
79signal_handler (int signo)
80{
81 int saved_errno = errno;
82
83 g_last_signal = signo;
84
85 /* Write signal number to pipe - write() is async-signal-safe */
86 char byte = (char)signo;
87 ssize_t ret = write (g_signal_pipe[1], &byte, 1);
88 (void)ret; /* Ignore - best effort in signal handler */
89
90 errno = saved_errno;
91}

108static int
109setup_signal_handling (void)
110{
111 /* Create self-pipe */
112 if (pipe (g_signal_pipe) < 0)
113 {
114 perror ("pipe");
115 return -1;
116 }
117
118 /* Make both ends non-blocking to prevent blocking in signal handler */
119 if (fcntl (g_signal_pipe[0], F_SETFL, O_NONBLOCK) < 0
120 || fcntl (g_signal_pipe[1], F_SETFL, O_NONBLOCK) < 0)
121 {
122 perror ("fcntl O_NONBLOCK");
123 close (g_signal_pipe[0]);
124 close (g_signal_pipe[1]);
125 return -1;
126 }
127
128 /* Set close-on-exec to prevent leaking to child processes */
129 fcntl (g_signal_pipe[0], F_SETFD, FD_CLOEXEC);
130 fcntl (g_signal_pipe[1], F_SETFD, FD_CLOEXEC);
131
132 /* Install signal handlers using sigaction (not signal()) */
133 struct sigaction sa;
134 memset (&sa, 0, sizeof (sa));
135 sa.sa_handler = signal_handler;
136 sigemptyset (&sa.sa_mask);
137 sa.sa_flags = SA_RESTART; /* Restart interrupted syscalls */
138
139 if (sigaction (SIGINT, &sa, NULL) < 0)
140 {
141 perror ("sigaction SIGINT");
142 return -1;
143 }
144
145 if (sigaction (SIGTERM, &sa, NULL) < 0)
146 {
147 perror ("sigaction SIGTERM");
148 return -1;
149 }
150
151 if (sigaction (SIGHUP, &sa, NULL) < 0)
152 {
153 perror ("sigaction SIGHUP");
154 return -1;
155 }
156
157 return 0;
158}

165static int
166check_signal_pipe (void)

168 char buf[16];

198static int
199handle_client_data (Socket_T client, SocketBuf_T inbuf, SocketBuf_T outbuf)
200{
201 char buf[BUFFER_SIZE];
202 ssize_t n;
203
204 (void)inbuf;
205 (void)outbuf;
206
207 TRY
208 {
209 n = Socket_recv (client, buf, sizeof (buf) - 1);
210 if (n > 0)
211 {
212 buf[n] = '\0';
213 printf ("[%s:%d] Received %zd bytes\n", Socket_getpeeraddr (client),
214 Socket_getpeerport (client), n);
215
216 /* Echo back */
217 Socket_sendall (client, buf, n);
218 return 1; /* Keep connection */
219 }
220 }
221 EXCEPT (Socket_Closed)
222 {
223 printf ("[%s:%d] Client disconnected\n", Socket_getpeeraddr (client),
224 Socket_getpeerport (client));
225 }
226 EXCEPT (Socket_Failed)
227 {
228 printf ("[%s:%d] Socket error\n", Socket_getpeeraddr (client),
229 Socket_getpeerport (client));
230 }
231 END_TRY;
232
233 return 0; /* Close connection */
234}

242main (int argc, char *argv[])
243{
244 /* Variables used across TRY/EXCEPT must be volatile to avoid clobbering */
245 volatile int port = DEFAULT_PORT;
246 Socket_T server = NULL;
247 SocketPoll_T poll = NULL;
248 SocketPool_T pool = NULL;
249 Arena_T arena = NULL;
250 int running = 1;
251
252 /* Parse command line */
253 if (argc > 1)
254 {
255 port = atoi (argv[1]);
256 if (port <= 0 || port > 65535)
257 {
258 fprintf (stderr, "Invalid port: %s\n", argv[1]);
259 return 1;
260 }
261 }
262
263 /* Setup signal handling infrastructure */
264 if (setup_signal_handling () < 0)
265 {
266 fprintf (stderr, "Failed to setup signal handling\n");
267 return 1;
268 }
269
270 printf ("=== Graceful Shutdown Example ===\n");
271 printf ("Press Ctrl+C or send SIGTERM to initiate graceful shutdown\n\n");
272
273 TRY
274 {
275 /* Create resources */
276 arena = Arena_new ();
277 server = Socket_new (AF_INET, SOCK_STREAM, 0);
278 poll = SocketPoll_new (MAX_CONNECTIONS + 10);
279 pool = SocketPool_new (arena, MAX_CONNECTIONS, BUFFER_SIZE);
280
281 /* Configure and bind server */
282 Socket_setreuseaddr (server);
283 Socket_bind (server, NULL, port);
284 Socket_listen (server, 128);
285 Socket_setnonblocking (server);
286
287 printf ("Server listening on port %d\n", port);
288 printf ("Drain timeout: %d seconds\n\n", DRAIN_TIMEOUT_MS / 1000);
289
290 /* Add server socket to poll */
291 SocketPoll_add (poll, server, POLL_READ, server);
292
293 /* Main event loop
294 *
295 * Note: We use a short poll timeout and check the signal pipe
296 * separately. This is because SocketPoll only supports Socket_T,
297 * not raw file descriptors. For more sophisticated integration,
298 * you could use poll()/select() directly for both.
299 */
300 while (running)
301 {
302 /* Check signal pipe first (non-blocking) */
303 if (check_signal_pipe ())
304 {
305 printf ("\n[SIGNAL] Received signal %d, initiating graceful "
306 "shutdown...\n",
307 (int)g_last_signal);
308 running = 0;
309 break;
310 }
311
312 /* Poll for socket events with short timeout */
313 SocketEvent_T *events;
314 int nevents = SocketPoll_wait (poll, &events, POLL_TIMEOUT_MS);
315
316 for (int i = 0; i < nevents && running; i++)
317 {
318 /* Check for new connections */
319 if (events[i].socket == server)
320 {
321 Socket_T client = Socket_accept (server);
322 if (client)
323 {
324 Socket_setnonblocking (client);
325 Connection_T conn = SocketPool_add (pool, client);
326 if (conn)
327 {
328 SocketPoll_add (poll, client, POLL_READ, conn);
329
330 /* Get pool statistics */
331 size_t active_count = SocketPool_get_active_count (pool);
332 size_t idle_count = SocketPool_get_idle_count (pool);
333 double hit_rate = SocketPool_get_hit_rate (pool);
334
335 printf ("[%s:%d] Client connected (active: %zu, idle: %zu, hit rate: %.1f%%)\n",
336 Socket_getpeeraddr (client),
337 Socket_getpeerport (client),
338 active_count, idle_count, hit_rate * 100.0);
339 }
340 else
341 {
342 printf ("Pool full, rejecting connection\n");
343 Socket_free (&client);
344 }
345 }
346 continue;
347 }
348
349 /* Handle client data */
350 Connection_T conn = events[i].data;
351 if (conn)
352 {
353 Socket_T client = Connection_socket (conn);
354 SocketBuf_T inbuf = Connection_inbuf (conn);
355 SocketBuf_T outbuf = Connection_outbuf (conn);
356
357 if (!handle_client_data (client, inbuf, outbuf))
358 {
359 /* Close connection */
360 SocketPoll_del (poll, client);
361 SocketPool_remove (pool, client);
362
363 /* Show socket statistics before closing */
364 SocketStats_T stats;
365 Socket_getstats (client, &stats);
366 printf ("[%s:%d] Stats: sent=%zu, recv=%zu, duration=%.1fs\n",
367 Socket_getpeeraddr (client),
368 Socket_getpeerport (client),
369 stats.bytes_sent, stats.bytes_received,
370 (Socket_get_monotonic_ms () - stats.create_time_ms) / 1000.0);
371
372 Socket_free (&client);
373 printf ("Connections remaining: %zu\n",
374 SocketPool_count (pool));
375 }
376 }
377 }
378 }
379
380 /* =========================================================================
381 * Graceful Shutdown Phase
382 * =========================================================================
383 */
384
385 printf ("\n=== Graceful Shutdown ===\n");
386 printf ("Active connections: %zu\n", SocketPool_count (pool));
387
388 /* Stop accepting new connections */
389 SocketPoll_del (poll, server);
390 Socket_free (&server);
391 server = NULL;
392 printf ("Stopped accepting new connections\n");
393
394 /* Initiate drain with timeout */
395 if (SocketPool_count (pool) > 0)
396 {
397 printf ("Draining %zu connections (timeout: %ds)...\n",
398 SocketPool_count (pool), DRAIN_TIMEOUT_MS / 1000);
399
400 SocketPool_drain (pool, DRAIN_TIMEOUT_MS);
401
402 /* Continue processing events until drain completes */
403 while (SocketPool_drain_poll (pool) > 0)
404 {
405 SocketEvent_T *events;
406 int timeout = SocketPool_drain_remaining_ms (pool);
407 if (timeout <= 0)
408 timeout = 100;
409
410 int nevents = SocketPoll_wait (poll, &events, timeout);
411
412 for (int i = 0; i < nevents; i++)
413 {
414 Connection_T conn = events[i].data;
415 if (conn)
416 {
417 Socket_T client = Connection_socket (conn);
418 SocketBuf_T inbuf = Connection_inbuf (conn);
419 SocketBuf_T outbuf = Connection_outbuf (conn);
420
421 /* Process remaining data, then close */
422 handle_client_data (client, inbuf, outbuf);
423
424 SocketPoll_del (poll, client);
425 SocketPool_remove (pool, client);
426 Socket_free (&client);
427
428 printf ("Drained connection (remaining: %zu)\n",
429 SocketPool_count (pool));
430 }
431 }
432 }
433
434 size_t remaining = SocketPool_count (pool);
435 if (remaining > 0)
436 {
437 printf ("Drain timeout - force closing %zu connections\n",
438 remaining);
439 SocketPool_drain_force (pool);
440 }
441 else
442 {
443 printf ("All connections drained gracefully\n");
444 }
445 }
446 else
447 {
448 printf ("No connections to drain\n");
449 }
450 }
451 EXCEPT (Socket_Failed)
452 {
453 fprintf (stderr, "Socket error\n"); /* Note: This returns const char * */
454 }
455 EXCEPT (SocketPoll_Failed)
456 {
457 fprintf (stderr, "Poll error\n");
458 }
459 EXCEPT (SocketPool_Failed)
460 {
461 fprintf (stderr, "Pool error\n");
462 }
463 END_TRY;
464
465 /* Cleanup */
466 printf ("\n=== Cleanup ===\n");
467
468 if (server)
469 {
470 Socket_free (&server);
471 printf ("Server socket closed\n");
472 }
473
474 if (pool)
475 {
476 SocketPool_free (&pool);
477 printf ("Connection pool freed\n");
478 }
479
480 if (poll)
481 {
482 SocketPoll_free (&poll);
483 printf ("Poll instance freed\n");
484 }
485
486 if (arena)
487 {
488 Arena_dispose (&arena);
489 printf ("Arena disposed\n");
490 }
491
492 cleanup_signal_handling ();
493 printf ("Signal handling cleaned up\n");
494
495 /* Print final statistics */
496 printf ("\n=== Final Statistics ===\n");
497 printf ("Peak connections: %d\n", SocketMetrics_get_peak_connections ());
498 printf ("Total sockets created: %d\n", SocketMetrics_get_socket_count ());
499
500 printf ("\nGraceful shutdown complete.\n");
501 return 0;
