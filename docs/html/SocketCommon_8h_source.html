639static inline int
640SocketCommon_check_bound_unix (const struct sockaddr_storage *addr)
641{
642 (void)addr; /* Suppress unused parameter warning */
643 return 1; /* Unix domain sockets are bound if getsockname succeeds */
644}

652static inline int
653SocketCommon_check_bound_by_family (const struct sockaddr_storage *addr)
654{
655 if (addr->ss_family == AF_INET)
656 return SocketCommon_check_bound_ipv4 (addr);
657 else if (addr->ss_family == AF_INET6)
658 return SocketCommon_check_bound_ipv6 (addr);
659 else if (addr->ss_family == AF_UNIX)
660 return SocketCommon_check_bound_unix (addr);
661 return 0;
662}
