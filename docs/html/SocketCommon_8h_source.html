475static inline int
476SocketCommon_check_bound_unix (const struct sockaddr_storage *addr)
477{
478 (void)addr; /* Suppress unused parameter warning */

487static inline int
488SocketCommon_check_bound_by_family (const struct sockaddr_storage *addr)
489{
490 if (addr->ss_family == AF_INET)
491 return SocketCommon_check_bound_ipv4 (addr);
492 else if (addr->ss_family == AF_INET6)
493 return SocketCommon_check_bound_ipv6 (addr);
494 else if (addr->ss_family == AF_UNIX)
495 return SocketCommon_check_bound_unix (addr);
496 return 0;
497}

525static inline void
526SocketLiveCount_increment (struct SocketLiveCount *tracker)
527{
528 pthread_mutex_lock (&tracker->mutex);
529 tracker->count++;
530 pthread_mutex_unlock (&tracker->mutex);
531}

537static inline void
538SocketLiveCount_decrement (struct SocketLiveCount *tracker)
539{
540 pthread_mutex_lock (&tracker->mutex);
541 if (tracker->count > 0)
542 tracker->count--;
543 pthread_mutex_unlock (&tracker->mutex);
544}

551static inline int
552SocketLiveCount_get (struct SocketLiveCount *tracker)
553{
554 int count;
555 pthread_mutex_lock (&tracker->mutex);

558 return count;
