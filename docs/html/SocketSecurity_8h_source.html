319static inline size_t
320SocketSecurity_safe_multiply (size_t a, size_t b)
321{
322 if (a == 0 || b == 0)
323 return 0;
324 if (a > SIZE_MAX / b)
325 return 0; /* Would overflow */
326 return a * b;
327}

339static inline size_t
340SocketSecurity_safe_add (size_t a, size_t b)
341{
342 if (a > SIZE_MAX - b)
343 return SIZE_MAX; /* Would overflow */
344 return a + b;
345}
