48/* ============================================================================
49 * Security Configuration Overview
50 * ============================================================================
51 *
52 * This section documents where security limits are defined. All limits can be
53 * overridden at compile time by defining them before including headers.
54 *
55 * CORE LIMITS (SocketConfig.h):
56 * SOCKET_MAX_CONNECTIONS - Max connections in pool (10000)
57 * SOCKET_MAX_BUFFER_SIZE - Max buffer per connection (1MB)
58 * SOCKET_MIN_BUFFER_SIZE - Min buffer size (512)
59 * SOCKET_MAX_POLL_EVENTS - Max events per poll (10000)
60 * SOCKET_MAX_LISTEN_BACKLOG - Max listen backlog (1024)
61 * ARENA_MAX_ALLOC_SIZE - Max arena allocation (see SOCKET_SECURITY_MAX_ALLOCATION, default 256MB)
62 *
63 * DNS LIMITS (SocketConfig.h):
64 * SOCKET_DNS_MAX_PENDING - Max pending DNS requests (1000)
65 * SOCKET_DNS_MAX_LABEL_LENGTH - Max DNS label length (63)
66 *
67 * RATE LIMITING (SocketConfig.h):
68 * SOCKET_RATELIMIT_DEFAULT_CONN_PER_SEC - Connection rate limit (100/s)
69 * SOCKET_RATELIMIT_DEFAULT_BURST - Burst capacity (50)
70 * SOCKET_RATELIMIT_DEFAULT_MAX_PER_IP - Per-IP connection limit (10)
71 *
72 * SYN FLOOD PROTECTION (SocketConfig.h):
73 * SOCKET_SYN_DEFAULT_WINDOW_MS - Rate measurement window (10s)
74 * SOCKET_SYN_DEFAULT_MAX_PER_WINDOW - Max attempts per IP (50)
75 * SOCKET_SYN_DEFAULT_GLOBAL_PER_SEC - Global rate limit (1000/s)
76 * SOCKET_SYN_DEFAULT_MAX_TRACKED_IPS - Max tracked IPs (100000)
77 *
78 * TIMEOUT DEFAULTS (SocketConfig.h):
79 * SOCKET_DEFAULT_CONNECT_TIMEOUT_MS - Connect timeout (30s)
80 * SOCKET_DEFAULT_DNS_TIMEOUT_MS - DNS timeout (5s)
81 * SOCKET_DEFAULT_IDLE_TIMEOUT - Idle timeout (5min)
82 * SOCKET_DEFAULT_POLL_TIMEOUT - Poll timeout (1s)
83 *
84 * HTTP CORE LIMITS (SocketHTTP.h):
85 * SOCKETHTTP_MAX_HEADER_NAME - Max header name (256)
86 * SOCKETHTTP_MAX_HEADER_VALUE - Max header value (8KB)
87 * SOCKETHTTP_MAX_HEADER_SIZE - Max total headers (64KB)
88 * SOCKETHTTP_MAX_HEADERS - Max header count (100)
89 * SOCKETHTTP_MAX_URI_LEN - Max URI length (8KB)
90 *
91 * HTTP/1.1 LIMITS (SocketHTTP1.h):
92 * SOCKETHTTP1_MAX_REQUEST_LINE - Max request line (8KB)
93 * SOCKETHTTP1_MAX_CHUNK_SIZE - Max chunk size (16MB)
94 * SOCKETHTTP1_MAX_TRAILER_SIZE - Max trailer size (4KB)
95 *
96 * HTTP/2 LIMITS (SocketHTTP2.h):
97 * SOCKETHTTP2_DEFAULT_MAX_CONCURRENT_STREAMS - Max streams (100)
98 * SOCKETHTTP2_DEFAULT_MAX_FRAME_SIZE - Max frame (16KB)
99 * SOCKETHTTP2_MAX_MAX_FRAME_SIZE - Max allowed frame (16MB)
100 * SOCKETHTTP2_DEFAULT_MAX_HEADER_LIST_SIZE - Max header list (16KB)
101 *
102 * HPACK LIMITS (SocketHPACK.h):
103 * SOCKETHPACK_MAX_TABLE_SIZE - Max dynamic table (64KB default)
104 *
105 * WEBSOCKET LIMITS (SocketWS-private.h):
106 * SOCKETWS_MAX_FRAME_SIZE - Max frame size (16MB)
107 * SOCKETWS_MAX_MESSAGE_SIZE - Max message size (64MB)
108 *
109 * TLS LIMITS (SocketTLSConfig.h):
110 * SOCKET_TLS_MAX_CERT_CHAIN_DEPTH - Max cert chain (10)
111 * SOCKET_TLS_MAX_ALPN_LEN - Max ALPN string (255)
112 * SOCKET_TLS_MAX_SNI_LEN - Max SNI hostname (255)
113 * SOCKET_TLS_MAX_PINS - Max cert pins (32)
114 * SOCKET_TLS_SESSION_CACHE_SIZE - Session cache (1000)
115 *
116 * PROXY LIMITS (SocketProxy.h):
117 * SOCKET_PROXY_MAX_HOSTNAME_LEN - Max proxy hostname (255)
118 * SOCKET_PROXY_MAX_USERNAME_LEN - Max proxy username (255)
119 * SOCKET_PROXY_MAX_PASSWORD_LEN - Max proxy password (255)

341static inline size_t
342SocketSecurity_safe_multiply (size_t a, size_t b)
343{
344 if (a == 0 || b == 0)
345 return 0;
346 if (a > SIZE_MAX / b)
347 return 0; /* Would overflow */
348 return a * b;
349}

361static inline size_t
362SocketSecurity_safe_add (size_t a, size_t b)
363{
364 if (a > SIZE_MAX - b)
365 return SIZE_MAX; /* Would overflow */
366 return a + b;
367}
