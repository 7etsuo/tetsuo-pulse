42/* ============================================================================
43 * Security Configuration Overview
44 * ============================================================================
45 *
46 * This section documents where security limits are defined. All limits can be
47 * overridden at compile time by defining them before including headers.
48 *
49 * CORE LIMITS (SocketConfig.h):
50 * SOCKET_MAX_CONNECTIONS - Max connections in pool (10000)
51 * SOCKET_MAX_BUFFER_SIZE - Max buffer per connection (1MB)
52 * SOCKET_MIN_BUFFER_SIZE - Min buffer size (512)
53 * SOCKET_MAX_POLL_EVENTS - Max events per poll (10000)
54 * SOCKET_MAX_LISTEN_BACKLOG - Max listen backlog (1024)
55 * ARENA_MAX_ALLOC_SIZE - Max arena allocation (see
56 * SOCKET_SECURITY_MAX_ALLOCATION, default 256MB)
57 *
58 * DNS LIMITS (SocketConfig.h):
59 * SOCKET_DNS_MAX_PENDING - Max pending DNS requests (1000)
60 * SOCKET_DNS_MAX_LABEL_LENGTH - Max DNS label length (63)
61 *
62 * RATE LIMITING (SocketConfig.h):
63 * SOCKET_RATELIMIT_DEFAULT_CONN_PER_SEC - Connection rate limit (100/s)
64 * SOCKET_RATELIMIT_DEFAULT_BURST - Burst capacity (50)
65 * SOCKET_RATELIMIT_DEFAULT_MAX_PER_IP - Per-IP connection limit (10)
66 *
67 * SYN FLOOD PROTECTION (SocketConfig.h):
68 * SOCKET_SYN_DEFAULT_WINDOW_MS - Rate measurement window (10s)
69 * SOCKET_SYN_DEFAULT_MAX_PER_WINDOW - Max attempts per IP (50)
70 * SOCKET_SYN_DEFAULT_GLOBAL_PER_SEC - Global rate limit (1000/s)
71 * SOCKET_SYN_DEFAULT_MAX_TRACKED_IPS - Max tracked IPs (100000)
72 *
73 * TIMEOUT DEFAULTS (SocketConfig.h):
74 * SOCKET_DEFAULT_CONNECT_TIMEOUT_MS - Connect timeout (30s)
75 * SOCKET_DEFAULT_DNS_TIMEOUT_MS - DNS timeout (5s)
76 * SOCKET_DEFAULT_IDLE_TIMEOUT - Idle timeout (5min)
77 * SOCKET_DEFAULT_POLL_TIMEOUT - Poll timeout (1s)
78 *
79 * HTTP CORE LIMITS (SocketHTTP.h):
80 * SOCKETHTTP_MAX_HEADER_NAME - Max header name (256)
81 * SOCKETHTTP_MAX_HEADER_VALUE - Max header value (8KB)
82 * SOCKETHTTP_MAX_HEADER_SIZE - Max total headers (64KB)
83 * SOCKETHTTP_MAX_HEADERS - Max header count (100)
84 * SOCKETHTTP_MAX_URI_LEN - Max URI length (8KB)
85 *
86 * HTTP/1.1 LIMITS (SocketHTTP1.h):
87 * SOCKETHTTP1_MAX_REQUEST_LINE - Max request line (8KB)
88 * SOCKETHTTP1_MAX_CHUNK_SIZE - Max chunk size (16MB)
89 * SOCKETHTTP1_MAX_TRAILER_SIZE - Max trailer size (4KB)
90 *
91 * HTTP/2 LIMITS (SocketHTTP2.h):
92 * SOCKETHTTP2_DEFAULT_MAX_CONCURRENT_STREAMS - Max streams (100)
93 * SOCKETHTTP2_DEFAULT_MAX_FRAME_SIZE - Max frame (16KB)
94 * SOCKETHTTP2_MAX_MAX_FRAME_SIZE - Max allowed frame (16MB)
95 * SOCKETHTTP2_DEFAULT_MAX_HEADER_LIST_SIZE - Max header list (16KB)
96 *
97 * HPACK LIMITS (SocketHPACK.h):
98 * SOCKETHPACK_MAX_TABLE_SIZE - Max dynamic table (64KB default)
99 *
100 * WEBSOCKET LIMITS (SocketWS-private.h):
101 * SOCKETWS_MAX_FRAME_SIZE - Max frame size (16MB)
102 * SOCKETWS_MAX_MESSAGE_SIZE - Max message size (64MB)
103 *
104 * TLS LIMITS (SocketTLSConfig.h):
105 * SOCKET_TLS_MAX_CERT_CHAIN_DEPTH - Max cert chain (10)
106 * SOCKET_TLS_MAX_ALPN_LEN - Max ALPN string (255)
107 * SOCKET_TLS_MAX_SNI_LEN - Max SNI hostname (255)
108 * SOCKET_TLS_MAX_PINS - Max cert pins (32)
109 * SOCKET_TLS_SESSION_CACHE_SIZE - Session cache (1000)
110 *
111 * PROXY LIMITS (SocketProxy.h):
112 * SOCKET_PROXY_MAX_HOSTNAME_LEN - Max proxy hostname (255)
113 * SOCKET_PROXY_MAX_USERNAME_LEN - Max proxy username (255)
114 * SOCKET_PROXY_MAX_PASSWORD_LEN - Max proxy password (255)
115 */

442static inline size_t
443SocketSecurity_safe_multiply (size_t a, size_t b)
444{
445 if (a == 0 || b == 0)
446 return 0;
447 if (a > SIZE_MAX / b)
448 return 0; /* Would overflow */
449 return a * b;
450}

465static inline size_t
466SocketSecurity_safe_add (size_t a, size_t b)
467{
468 if (a > SIZE_MAX - b)
469 return SIZE_MAX; /* Would overflow */
470 return a + b;
471}
