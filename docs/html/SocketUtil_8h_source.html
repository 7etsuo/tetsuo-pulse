1733static inline int
1734SocketTimeout_poll_timeout (int current_timeout_ms, int64_t deadline_ms)
1735{

1737
1738 if (deadline_ms == 0)
1739 return current_timeout_ms; /* No deadline */
1740
1741 remaining = SocketTimeout_remaining_ms (deadline_ms);
1742 if (remaining == 0)
1743 return 0; /* Already expired */
1744

1747
