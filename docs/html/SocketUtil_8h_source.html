1220static inline int
1221SocketTimeout_poll_timeout (int current_timeout_ms, int64_t deadline_ms)
1222{
1223 int64_t remaining;
1224
1225 if (deadline_ms == 0)
1226 return current_timeout_ms; /* No deadline */
1227
1228 remaining = SocketTimeout_remaining_ms (deadline_ms);
1229 if (remaining == 0)
1230 return 0; /* Already expired */
1231
1232 if (remaining == -1)
1233 return current_timeout_ms; /* No deadline (shouldn't happen here) */
1234
1235 /* Cap remaining to INT_MAX for poll() */

1237 remaining = INT_MAX;
1238
1239 /* Return minimum of current timeout and remaining */
1240 if (current_timeout_ms < 0)
1241 return (int)remaining;
1242
1243 return (current_timeout_ms < (int)remaining) ? current_timeout_ms
1244 : (int)remaining;
