339#define SOCKET_ERROR_APPLY_TRUNCATION(ret) \
340 do \
341 { \
342 if ((ret) >= (int)SOCKET_ERROR_BUFSIZE) \
343 { \
344 socket_error_buf[SOCKET_ERROR_BUFSIZE - 1] = '\0'; \
345 if (SOCKET_ERROR_BUFSIZE >= SOCKET_ERROR_TRUNCATION_SIZE + 1) \
346 { \
347 memcpy (socket_error_buf + SOCKET_ERROR_BUFSIZE \
348 - SOCKET_ERROR_TRUNCATION_SIZE, \
349 SOCKET_ERROR_TRUNCATION_MARKER, \
350 SOCKET_ERROR_TRUNCATION_SIZE - 1); \
351 socket_error_buf[SOCKET_ERROR_BUFSIZE - 1] = '\0'; \
352 } \
353 } \
354 } \
355 while (0)

361#define SOCKET_ERROR_FMT(fmt, ...) \
362 do \
363 { \
364 socket_last_errno = errno; \
365 int _socket_error_ret = snprintf ( \
366 socket_error_buf, SOCKET_ERROR_BUFSIZE, fmt " (errno: %d - %s)", \
367 ##__VA_ARGS__, socket_last_errno, \
368 Socket_safe_strerror (socket_last_errno)); \
369 SOCKET_ERROR_APPLY_TRUNCATION (_socket_error_ret); \
370 (void)_socket_error_ret; \
371 SocketLog_emit (SOCKET_LOG_ERROR, SOCKET_LOG_COMPONENT, \
372 socket_error_buf); \
373 } \
374 while (0)
