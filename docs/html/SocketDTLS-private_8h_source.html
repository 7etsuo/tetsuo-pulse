611static inline DTLSHandshakeState
612dtls_handle_ssl_error (SocketDgram_T socket, SSL *ssl, int ssl_result)
613{
614 int ssl_error = SSL_get_error (ssl, ssl_result);
615
616 switch (ssl_error)
617 {
618 case SSL_ERROR_NONE:
619 socket->dtls_handshake_done = 1;
620 return DTLS_HANDSHAKE_COMPLETE;
621
622 case SSL_ERROR_WANT_READ:
623 socket->dtls_handshake_done = 0;
624 errno = EAGAIN;
625 return DTLS_HANDSHAKE_WANT_READ;
626
627 case SSL_ERROR_WANT_WRITE:
628 socket->dtls_handshake_done = 0;
629 errno = EAGAIN;
630 return DTLS_HANDSHAKE_WANT_WRITE;
631
632 default:
633 socket->dtls_handshake_done = 0;
634 return DTLS_HANDSHAKE_ERROR;
635 }
636}

811static inline int
812dtls_validate_file_path (const char *path)
813{
814 if (!path || !*path)
815 return 0;
816
817 size_t len = strlen (path);
818 if (len == 0 || len > SOCKET_DTLS_MAX_PATH_LEN)
819 return 0;
820
821 /* Check for path traversal sequences (prefer absolute paths) */
822 if (path[0] != '/')
823 { /* Optional: warn or reject relative; preference for absolute */
824 /* Relative paths accepted but may be insecure if cwd untrusted */
825 }
826 if (strstr (path, "../") != NULL || strstr (path, "..\\") != NULL
827 || strstr (path, "/..") != NULL || strstr (path, "\\..") != NULL
828 || strstr (path, "..") != NULL)
829 {
830 return 0;
831 }
832
833 /* Check for control characters */
834 for (size_t i = 0; i < len; i++)
835 {
836 unsigned char c = (unsigned char)path[i];
837 if (c < 32 || c == 127)
838 return 0;
839 }
840
841 return 1;
842}
