662static inline DTLSHandshakeState
663dtls_handle_ssl_error (SocketDgram_T socket, SSL *ssl, int ssl_result)
664{
665 int ssl_error = SSL_get_error (ssl, ssl_result);
666
667 switch (ssl_error)
668 {
669 case SSL_ERROR_NONE:
670 socket->dtls_handshake_done = 1;
671 return DTLS_HANDSHAKE_COMPLETE;
672
673 case SSL_ERROR_WANT_READ:
674 socket->dtls_handshake_done = 0;
675 errno = EAGAIN;
676 return DTLS_HANDSHAKE_WANT_READ;
677
678 case SSL_ERROR_WANT_WRITE:
679 socket->dtls_handshake_done = 0;
680 errno = EAGAIN;
681 return DTLS_HANDSHAKE_WANT_WRITE;
682
683 default:
684 socket->dtls_handshake_done = 0;
685 return DTLS_HANDSHAKE_ERROR;
686 }
687}

887static inline int
888dtls_validate_file_path (const char *path)
889{
890 if (!path || !*path)
891 return 0;
892
893 size_t len = strlen (path);
894 if (len == 0 || len > SOCKET_DTLS_MAX_PATH_LEN)
895 return 0;
896
897 /* Check for path traversal sequences (prefer absolute paths) */
898 if (path[0] != '/')
899 { /* Optional: warn or reject relative; preference for absolute */
900 /* Relative paths accepted but may be insecure if cwd untrusted */
901 }
902 if (strstr (path, "../") != NULL || strstr (path, "..\\") != NULL
903 || strstr (path, "/..") != NULL || strstr (path, "\\..") != NULL
904 || strstr (path, "..") != NULL)
905 {
906 return 0;
907 }
908
909 /* Check for control characters */
910 for (size_t i = 0; i < len; i++)
911 {
912 unsigned char c = (unsigned char)path[i];
913 if (c < 32 || c == 127)
914 return 0;
915 }
916
917 return 1;
918}
