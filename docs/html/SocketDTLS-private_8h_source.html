215static inline DTLSHandshakeState
216dtls_handle_ssl_error (SocketDgram_T socket, SSL *ssl, int ssl_result)
217{
218 int ssl_error = SSL_get_error (ssl, ssl_result);
219
220 switch (ssl_error)
221 {
222 case SSL_ERROR_NONE:
223 socket->dtls_handshake_done = 1;
224 return DTLS_HANDSHAKE_COMPLETE;
225
226 case SSL_ERROR_WANT_READ:
227 socket->dtls_handshake_done = 0;
228 errno = EAGAIN;
229 return DTLS_HANDSHAKE_WANT_READ;
230
231 case SSL_ERROR_WANT_WRITE:
232 socket->dtls_handshake_done = 0;
233 errno = EAGAIN;
234 return DTLS_HANDSHAKE_WANT_WRITE;
235
236 default:
237 socket->dtls_handshake_done = 0;
238 return DTLS_HANDSHAKE_ERROR;
239 }
240}

292static inline int
293dtls_validate_file_path (const char *path)
294{
295 if (!path || !*path)
296 return 0;
297
298 size_t len = strlen (path);
299 if (len == 0 || len > SOCKET_DTLS_MAX_PATH_LEN)
300 return 0;
301
302 /* Check for path traversal sequences (prefer absolute paths) */
303 if (path[0] != '/')
304 { /* Optional: warn or reject relative; preference for absolute */
305 /* Relative paths accepted but may be insecure if cwd untrusted */
306 }
307 if (strstr (path, "../") != NULL || strstr (path, "..\\") != NULL
308 || strstr (path, "/..") != NULL || strstr (path, "\\..") != NULL
309 || strstr (path, "..") != NULL)
310 {
311 return 0;
312 }
313
314 /* Check for control characters */
315 for (size_t i = 0; i < len; i++)
316 {
317 unsigned char c = (unsigned char)path[i];
318 if (c < 32 || c == 127)
319 return 0;
320 }
321
322 return 1;
323}
