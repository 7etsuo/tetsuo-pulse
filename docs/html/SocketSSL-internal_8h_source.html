86static inline int
87ssl_contains_path_traversal (const char *path, size_t len)
88{
89 /* Reject any path containing ".." - simple but comprehensive
90 * This catches all traversal attempts including:
91 * - /../ (Unix traversal)
92 * - \..\ (Windows traversal)
93 * - /.. or \.. at end
94 * - ../ or ..\ at start
95 * - Just ".." alone
96 * - Mixed separators: /..\ or \../
97 */
98 if (strstr (path, "..") != NULL)
99 return 1;
100
101 (void)len; /* Reserved for future validation */
102 return 0;
103}

121static inline int
122ssl_contains_control_chars (const char *path, size_t len)
123{
124 for (size_t i = 0; i < len; i++)
125 {
126 unsigned char c = (unsigned char)path[i];
127 /* ASCII control: 0x00-0x1F (space-1) and 0x7F (DEL) */
128 if (c < 0x20 || c == 0x7F)
129 return 1;
130 }
131 return 0;
132}

199static inline int
200ssl_validate_file_path (const char *path, size_t max_len)
201{
202 /* NULL or empty path */
203 if (!path || !*path)
204 return 0;
205
206 size_t len = strlen (path);
207
208 /* Length validation (also catches empty after strlen) */
209 if (len == 0 || len > max_len)
210 return 0;
211
212 /* Control character check (includes embedded nulls via strlen limitation)
213 * Must check BEFORE any string operations that might be confused by special
214 * chars */
215 if (ssl_contains_control_chars (path, len))
216 return 0;
217
218 /* Path traversal detection - reject ANY ".." sequence
219 * This is defense-in-depth; realpath() in callers provides additional
220 * protection */
221 if (ssl_contains_path_traversal (path, len))
222 return 0;
223
224 /* Symlink detection via lstat() - reject symlinks to prevent attacks
225 * lstat() doesn't follow symlinks, so we can detect them directly */
226 struct stat sb;
227 if (lstat (path, &sb) == 0)
228 {
229 if (S_ISLNK (sb.st_mode))
230 return 0; /* Reject symlinks */
231 }
232 /* lstat() failure (ENOENT, EACCES, etc.) is OK - validation passes,
233 * but actual file operations will fail with appropriate errors */
234
235 return 1;
236}

323static inline void
324ssl_format_openssl_error_to_buf (const char *context, char *buf,
325 size_t buf_size)
326{
327 unsigned long err;
328 char err_str[SOCKET_SSL_OPENSSL_ERRSTR_BUFSIZE];
329
330 /* Validate output buffer parameters */
331 if (!buf || buf_size == 0)
332 {
333 ERR_clear_error ();
334 return;
335 }
336
337 /* ERR_get_error() returns 0 if no error is queued.
338 * It removes the error from the queue as a side effect. */
339 err = ERR_get_error ();
340
341 if (err != 0)
342 {
343 /* ERR_error_string_n() safely formats into fixed-size buffer.
344 * It null-terminates and truncates if necessary. */
345 ERR_error_string_n (err, err_str, sizeof (err_str));
346
347 /* Format with context prefix */
348 if (context && *context)
349 {
350 snprintf (buf, buf_size, "%s: %s", context, err_str);
351 }
352 else
353 {
354 snprintf (buf, buf_size, "%s", err_str);
355 }
356 }
357 else
358 {
359 /* No error in queue - provide meaningful fallback */
360 if (context && *context)
361 {
362 snprintf (buf, buf_size, "%s: Unknown error (no OpenSSL error code)",
363 context);
364 }
365 else
366 {
367 snprintf (buf, buf_size, "Unknown error (no OpenSSL error code)");
368 }
369 }
370
371 /* CRITICAL: Clear the entire error queue to prevent:
372 * 1. Stale errors affecting subsequent unrelated operations
373 * 2. Memory buildup from unread errors
374 * 3. Error leakage between different logical operations
375 *
376 * Per OpenSSL documentation: "After handling an error, the error queue
377 * should be cleared using ERR_clear_error()" */
378 ERR_clear_error ();
379}
