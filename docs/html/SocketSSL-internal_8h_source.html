90static inline int
91ssl_validate_file_path (const char *path, size_t max_len)
92{
93 if (!path || !*path)
94 return 0;
95
96 size_t len = strlen (path);
97 if (len == 0 || len > max_len)
98 return 0;
99
100 /* Check for specific path traversal sequences (avoid false positives on
101 * filenames like "cert..pem") */
102 const char *traversal_patterns[]
103 = { "/../", "\\..\\", "/..\\",
104 "\\../", "/.../", "\\../", /* Added context-aware */
105 NULL };
106 for (const char **pat = traversal_patterns; *pat != NULL; ++pat)
107 {
108 if (strstr (path, *pat) != NULL)
109 return 0;
110 }
111
112 /* Reject paths starting with relative traversal */
113 if (strncmp (path, "../", 3) == 0 || strncmp (path, "..\\", 3) == 0)
114 return 0;
115
116 /* Additional symlink detection (optional, conservative: reject if detectable
117 * symlink) */
118 struct stat sb;
119 if (lstat (path, &sb) == 0)
120 {
121 if (S_ISLNK (sb.st_mode))
122 return 0; /* Reject symlinks to prevent attacks */
123 }
124 /* If lstat fails (e.g., no perm), continue validation (false negative ok for
125 * usability) */
126
127 /* Check for control characters (ASCII 0-31 and 127) */
128 for (size_t i = 0; i < len; i++)
129 {
130 unsigned char c = (unsigned char)path[i];
131 if (c < 32 || c == 127)
132 return 0;
133 }
134
135 return 1;
136}

166static inline void
167ssl_format_openssl_error_to_buf (const char *context, char *buf,
168 size_t buf_size)
169{
170 unsigned long err = ERR_get_error ();
171 char err_str[SOCKET_SSL_OPENSSL_ERRSTR_BUFSIZE];
172
173 if (err != 0)
174 {
175 ERR_error_string_n (err, err_str, sizeof (err_str));
176 snprintf (buf, buf_size, "%s: %s", context, err_str);
177 }
178 else
179 {
180 snprintf (buf, buf_size, "%s: Unknown error", context);
181 }
182
183 /* Clear remaining errors to prevent stale error information from
184 * affecting subsequent operations or leaking to callers */
185 ERR_clear_error ();
186}
