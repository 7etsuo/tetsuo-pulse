1460static inline int
1461ws_is_valid_close_code (int code)
1462{
1463 /* RFC 6455 Section 7.4.1 */
1464 if (code < 1000)
1465 return 0;
1466 /* code >= 1000 is implied here since we didn't return above */
1467 if (code <= 1003)
