691static inline int64_t
692socketproxy_get_time_ms (void)
693{
694 struct timespec ts;
695
696 if (clock_gettime (CLOCK_MONOTONIC, &ts) < 0)
697 return 0;
698
699 return (int64_t)ts.tv_sec * SOCKET_PROXY_MS_PER_SEC
700 + (int64_t)ts.tv_nsec / SOCKET_PROXY_NS_PER_MS;
701}

717static inline int64_t
718socketproxy_elapsed_ms (int64_t start_ms)
719{
720 int64_t elapsed = socketproxy_get_time_ms () - start_ms;
721 return (elapsed < 0) ? 0 : elapsed;
722}
