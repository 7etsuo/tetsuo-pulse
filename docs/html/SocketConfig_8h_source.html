390/* Safe system call wrappers
391 *
392 * SAFE_CLOSE: Close file descriptor with proper EINTR handling
393 *
394 * Per POSIX.1-2008: Do NOT retry close() on EINTR. The file descriptor
395 * state is unspecified after close() returns with EINTR - it may or may
396 * not be closed. Retrying could close a different FD if the descriptor
397 * was reused by another thread. We treat EINTR as success (don't log error)
398 * since the FD is likely closed anyway.
399 *
400 * Reference: POSIX.1-2008, close() specification, Application Usage
401 */
402#define SAFE_CLOSE(fd) \
403 do \
404 { \
405 if ((fd) >= 0) \
406 { \
407 int _safe_close_result = close(fd); \
408 if (_safe_close_result < 0 && errno != EINTR) \
409 { \
410 /* Log error but don't fail - fd is closed anyway */ \
411 perror("close"); \
412 } \
413 /* EINTR is silently treated as success - FD is likely closed */ \
414 } \
415 } while (0)
