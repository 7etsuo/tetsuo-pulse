35/* ============================================================================
36 * Custom Certificate Lookup Callback Example
37 * ============================================================================
38 *
39 * This demonstrates how to implement a custom certificate lookup for HSM,
40 * database, or other non-filesystem sources. In this example, we just log
41 * the lookup request and return NULL (no certificate found).
42 */
43
44typedef struct
45{
46 const char *source_name;
47 int lookup_count;
48} CertLookupContext;

50static X509 *
51example_cert_lookup (X509_STORE_CTX *store_ctx, const X509_NAME *name,
52 void *user_data)
53{
54 CertLookupContext *ctx = (CertLookupContext *)user_data;
55 (void)store_ctx; /* Unused in this example */
56
57 ctx->lookup_count++;
58
59 /* Log the lookup request.
60 * X509_NAME_oneline takes non-const but doesn't modify, so cast is safe. */
61 char name_buf[256];
62 X509_NAME_oneline ((X509_NAME *)name, name_buf, sizeof (name_buf));
63 printf (" üìã Certificate lookup #%d from '%s':\n", ctx->lookup_count,
64 ctx->source_name);
65 printf (" Subject: %s\n", name_buf);
66
67 /* In a real implementation, you would:
68 * 1. Query your HSM/database for a certificate with this subject
69 * 2. Parse the certificate data into an X509 object
70 * 3. Return the X509 (caller takes ownership via X509_free)
71 *
72 * Example for HSM:
73 * unsigned char *cert_der;
74 * size_t cert_len;
75 * if (hsm_find_cert_by_subject(session, name_buf, &cert_der, &cert_len)) {
76 * const unsigned char *p = cert_der;
77 * X509 *cert = d2i_X509(NULL, &p, cert_len);
78 * free(cert_der);
79 * return cert; // Ownership transferred
80 * }
81 *
82 * For now, return NULL to indicate no certificate found.
83 */
84 printf (" Result: Not found (example callback)\n");
85 return NULL;
86}
