116 *
117 * @brief Thread-local exception for detailed error messages.
118 * @ingroup connection_mgmt
119 *
120 * Thread-local exception for detailed error messages across all SocketPool
121 * implementation files. Uses the centralized error buffer (socket_error_buf)
122 * from SocketUtil.h for consistent error formatting.
123 *
124 * Benefits:
125 * - Single thread-local error buffer (socket_error_buf) for all modules
126 * - Consistent error formatting with SOCKET_ERROR_FMT/MSG macros
127 * - Thread-safe exception raising
128 * - Automatic logging integration via SocketLog_emit
129 *
130 * NOTE: For multi-file modules like SocketPool, we use an extern declaration
131 * here and the actual definition in SocketPool-core.c. This allows all
132 * implementation files to share the same thread-local exception variable.
133 */
134
146#ifdef _WIN32
147extern __declspec (thread) Except_T SocketPool_DetailedException;

149extern __thread Except_T SocketPool_DetailedException;
150#endif

164#define RAISE_POOL_ERROR(exception) \
165 do \
