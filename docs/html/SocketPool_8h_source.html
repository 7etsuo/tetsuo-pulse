845/* ============================================================================
846 * Graceful Shutdown (Drain) API
847 * ============================================================================
848 *
849 * @brief Industry-standard graceful shutdown following patterns from nginx, HAProxy,
850 * and Go http.Server. Provides clean state machine transitions, non-blocking
851 * APIs for event loop integration, and timeout-guaranteed completion.
852 *
853 * @ingroup connection_mgmt
854 *
855 * State Machine:
856 * drain(timeout)
857 * +---------+ ───────────────> +----------+
858 * | RUNNING | | DRAINING |
859 * +---------+ +----------+
860 * ^ │
861 * │ │ (count == 0 OR timeout)
862 * │ +----------+ │
863 * +────────────| STOPPED |<─────────+
864 * (restart) +----------+
865 *
866 * Typical usage:
867 * SocketPool_drain(pool, 30000); // Start 30s drain
868 * while (SocketPool_drain_poll(pool) > 0) {
869 * // Continue event loop, connections closing naturally
870 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
871 * }
872 * SocketPool_free(&pool);
873 *
874 * @see SocketPool_drain() to initiate drain.
875 * @see SocketPool_drain_poll() for non-blocking completion.
876 * @see SocketPool_drain_wait() for blocking completion.
877 */
