557/* ============================================================================
558 * Graceful Shutdown (Drain) API
559 * ============================================================================
560 *
561 * Industry-standard graceful shutdown following patterns from nginx, HAProxy,
562 * and Go http.Server. Provides clean state machine transitions, non-blocking
563 * APIs for event loop integration, and timeout-guaranteed completion.
564 *
565 * State Machine:
566 * drain(timeout)
567 * +---------+ ───────────────> +----------+
568 * | RUNNING | | DRAINING |
569 * +---------+ +----------+
570 * ^ │
571 * │ │ (count == 0 OR timeout)
572 * │ +----------+ │
573 * +────────────| STOPPED |<─────────+
574 * (restart) +----------+
575 *
576 * Typical usage:
577 * SocketPool_drain(pool, 30000); // Start 30s drain
578 * while (SocketPool_drain_poll(pool) > 0) {
579 * // Continue event loop, connections closing naturally
580 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
581 * }
582 * SocketPool_free(&pool);
583 */
584
