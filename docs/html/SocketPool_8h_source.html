857/* ============================================================================
858 * Graceful Shutdown (Drain) API
859 * ============================================================================
860 *
861 * @brief Industry-standard graceful shutdown following patterns from nginx, HAProxy,
862 * and Go http.Server. Provides clean state machine transitions, non-blocking
863 * APIs for event loop integration, and timeout-guaranteed completion.
864 *
865 * @ingroup connection_mgmt
866 *
867 * State Machine:
868 * drain(timeout)
869 * +---------+ ───────────────> +----------+
870 * | RUNNING | | DRAINING |
871 * +---------+ +----------+
872 * ^ │
873 * │ │ (count == 0 OR timeout)
874 * │ +----------+ │
875 * +────────────| STOPPED |<─────────+
876 * (restart) +----------+
877 *
878 * Typical usage:
879 * SocketPool_drain(pool, 30000); // Start 30s drain
880 * while (SocketPool_drain_poll(pool) > 0) {
881 * // Continue event loop, connections closing naturally
882 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
883 * }
884 * SocketPool_free(&pool);
885 *
886 * @see SocketPool_drain() to initiate drain.
887 * @see SocketPool_drain_poll() for non-blocking completion.
888 * @see SocketPool_drain_wait() for blocking completion.
889 */
