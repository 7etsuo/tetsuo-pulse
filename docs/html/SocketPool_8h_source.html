1098/* ============================================================================
1099 * Graceful Shutdown (Drain) API
1100 * ============================================================================
1101 *
1102 * @brief Industry-standard graceful shutdown following patterns from nginx,
1103 * HAProxy, and Go http.Server. Provides clean state machine transitions,
1104 * non-blocking APIs for event loop integration, and timeout-guaranteed
1105 * completion.
1106 *
1107 * @ingroup connection_mgmt
1108 *
1109 * State Machine:
1110 * drain(timeout)
1111 * +---------+ ───────────────> +----------+
1112 * | RUNNING | | DRAINING |
1113 * +---------+ +----------+
1114 * ^ │
1115 * │ │ (count == 0 OR timeout)
1116 * │ +----------+ │
1117 * +────────────| STOPPED |<─────────+
1118 * (restart) +----------+
1119 *
1120 * Typical usage:
1121 * SocketPool_drain(pool, 30000); // Start 30s drain
1122 * while (SocketPool_drain_poll(pool) > 0) {
1123 * // Continue event loop, connections closing naturally
1124 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
1125 * }
1126 * SocketPool_free(&pool);
1127 *
1128 * @see SocketPool_drain() to initiate drain.
1129 * @see SocketPool_drain_poll() for non-blocking completion.
1130 * @see SocketPool_drain_wait() for blocking completion.
1131 */
