981/* ============================================================================
982 * Graceful Shutdown (Drain) API
983 * ============================================================================
984 *
985 * @brief Industry-standard graceful shutdown following patterns from nginx,
986 * HAProxy, and Go http.Server. Provides clean state machine transitions,
987 * non-blocking APIs for event loop integration, and timeout-guaranteed
988 * completion.
989 *
990 * @ingroup connection_mgmt
991 *
992 * State Machine:
993 * drain(timeout)
994 * +---------+ ───────────────> +----------+
995 * | RUNNING | | DRAINING |
996 * +---------+ +----------+
997 * ^ │
998 * │ │ (count == 0 OR timeout)
999 * │ +----------+ │
1000 * +────────────| STOPPED |<─────────+
1001 * (restart) +----------+
1002 *
1003 * Typical usage:
1004 * SocketPool_drain(pool, 30000); // Start 30s drain
1005 * while (SocketPool_drain_poll(pool) > 0) {
1006 * // Continue event loop, connections closing naturally
1007 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
1008 * }
1009 * SocketPool_free(&pool);
1010 *
1011 * @see SocketPool_drain() to initiate drain.
1012 * @see SocketPool_drain_poll() for non-blocking completion.
1013 * @see SocketPool_drain_wait() for blocking completion.
1014 */
