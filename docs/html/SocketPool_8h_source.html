954/* ============================================================================
955 * Graceful Shutdown (Drain) API
956 * ============================================================================
957 *
958 * @brief Industry-standard graceful shutdown following patterns from nginx, HAProxy,
959 * and Go http.Server. Provides clean state machine transitions, non-blocking
960 * APIs for event loop integration, and timeout-guaranteed completion.
961 *
962 * @ingroup connection_mgmt
963 *
964 * State Machine:
965 * drain(timeout)
966 * +---------+ ───────────────> +----------+
967 * | RUNNING | | DRAINING |
968 * +---------+ +----------+
969 * ^ │
970 * │ │ (count == 0 OR timeout)
971 * │ +----------+ │
972 * +────────────| STOPPED |<─────────+
973 * (restart) +----------+
974 *
975 * Typical usage:
976 * SocketPool_drain(pool, 30000); // Start 30s drain
977 * while (SocketPool_drain_poll(pool) > 0) {
978 * // Continue event loop, connections closing naturally
979 * SocketPoll_wait(poll, &events, SocketPool_drain_remaining_ms(pool));
980 * }
981 * SocketPool_free(&pool);
982 *
983 * @see SocketPool_drain() to initiate drain.
984 * @see SocketPool_drain_poll() for non-blocking completion.
985 * @see SocketPool_drain_wait() for blocking completion.
986 */
