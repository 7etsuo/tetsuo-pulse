559static double
560reconnect_jitter (void)
561{
562 unsigned int value;
563 if (SocketCrypto_random_bytes (&value, sizeof (value)) == 0)
564 {
565 return (double)value / (double)0xFFFFFFFFU;
566 }
567 else
568 {
569 /* Fallback to time-based PRNG */
570#ifdef _WIN32
571 static __declspec (thread) unsigned int seed = 0;
572#else
573 static __thread unsigned int seed = 0;
574#endif
575 if (seed == 0)
576 {
577 seed = (unsigned int)Socket_get_monotonic_ms ();
578 }
579 /* xorshift32 */
580 seed ^= seed << 13;
581 seed ^= seed >> 17;
582 seed ^= seed << 5;
583 return (double)seed / (double)0xFFFFFFFFU;
584 }
585}
