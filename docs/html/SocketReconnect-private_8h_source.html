521static double
522reconnect_jitter (void)
523{
524 unsigned int value;
525 if (SocketCrypto_random_bytes (&value, sizeof (value)) == 0)
526 {
527 return (double)value / (double)0xFFFFFFFFU;
528 }
529 else
530 {
531 /* Fallback to time-based PRNG */
532#ifdef _WIN32
533 static __declspec (thread) unsigned int seed = 0;
534#else
535 static __thread unsigned int seed = 0;
536#endif
537 if (seed == 0)
538 {
539 seed = (unsigned int)Socket_get_monotonic_ms ();
540 }
541 /* xorshift32 */
542 seed ^= seed << 13;
543 seed ^= seed >> 17;
544 seed ^= seed << 5;
545 return (double)seed / (double)0xFFFFFFFFU;
546 }
