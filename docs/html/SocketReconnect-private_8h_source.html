189static inline double
190socketreconnect_random_double(void)
191{
192 unsigned int value;
193 if (SocketCrypto_random_bytes(&value, sizeof(value)) == 0) {
194 return (double)value / (double)0xFFFFFFFFU;
195 } else {
196 /* Fallback to time-based PRNG */
197#ifdef _WIN32
198 static __declspec(thread) unsigned int seed = 0;
199#else
200 static __thread unsigned int seed = 0;
201#endif
202 if (seed == 0) {
203 seed = (unsigned int)Socket_get_monotonic_ms();
204 }
205 /* xorshift32 */
206 seed ^= seed << 13;
207 seed ^= seed >> 17;
208 seed ^= seed << 5;
209 return (double)seed / (double)0xFFFFFFFFU;
210 }
211}
