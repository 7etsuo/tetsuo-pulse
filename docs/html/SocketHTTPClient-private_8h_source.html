41/* ============================================================================
42 * Exception Handling (Centralized)
43 * ============================================================================
44 *
45 * REFACTOR: Uses centralized exception infrastructure from SocketUtil.h
46 * instead of module-specific thread-local buffers and exception copies.
47 *
48 * Benefits:
49 * - Single thread-local error buffer (socket_error_buf) for all modules
50 * - Consistent error formatting with SOCKET_ERROR_FMT/MSG macros
51 * - Thread-safe exception raising via SOCKET_RAISE_MODULE_ERROR
52 * - Automatic logging integration via SocketLog_emit
53 *
54 * The thread-local exception (HTTPClient_DetailedException) is declared
55 * in SocketHTTPClient.c using SOCKET_DECLARE_MODULE_EXCEPTION(HTTPClient).
56 */

764static inline unsigned
765httpclient_host_hash (const char *host, int port, size_t table_size)
766{
767 size_t host_len = strlen (host);
768 unsigned host_hash
769 = socket_util_hash_djb2_ci_len (host, host_len, table_size);
770 unsigned port_hash = socket_util_hash_uint ((unsigned)port, table_size);
771 unsigned combined = host_hash ^ port_hash;
772 return socket_util_hash_uint (combined, table_size);
773}
