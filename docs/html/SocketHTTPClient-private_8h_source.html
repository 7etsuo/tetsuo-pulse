857static inline unsigned
858httpclient_host_hash (const char *host, int port, size_t table_size)
859{
860 size_t host_len = strlen (host);
861 unsigned host_hash
862 = socket_util_hash_djb2_ci_len (host, host_len, table_size);
863 unsigned port_hash = socket_util_hash_uint ((unsigned)port, table_size);
864 unsigned combined = host_hash ^ port_hash;
865 return socket_util_hash_uint (combined, table_size);
866}
