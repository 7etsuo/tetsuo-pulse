223static inline TLSHandshakeState
224tls_handle_ssl_error (Socket_T socket, SSL *ssl, int ssl_result)
225{
226 int ssl_error = SSL_get_error (ssl, ssl_result);
227
228 switch (ssl_error)
229 {
230 case SSL_ERROR_NONE:
231 socket->tls_handshake_done = 1;
232 return TLS_HANDSHAKE_COMPLETE;
233
234 case SSL_ERROR_WANT_READ:
235 socket->tls_handshake_done = 0;
236 errno = EAGAIN;
237 return TLS_HANDSHAKE_WANT_READ;
238
239 case SSL_ERROR_WANT_WRITE:
240 socket->tls_handshake_done = 0;
241 errno = EAGAIN;
242 return TLS_HANDSHAKE_WANT_WRITE;
243
244 case SSL_ERROR_ZERO_RETURN:
245 /* Clean shutdown by peer - not an error per se, but connection is done
246 */
247 socket->tls_handshake_done = 0;
248 return TLS_HANDSHAKE_ERROR;
249
250 case SSL_ERROR_SYSCALL:
251 /* System call error - errno contains the actual error.
252 * If errno is 0, it typically means unexpected EOF (connection reset).
253 * Do NOT overwrite errno here - preserve it for caller diagnostics. */
254 socket->tls_handshake_done = 0;
255 if (errno == 0)
256 errno = ECONNRESET; /* Unexpected EOF treated as connection reset */
257 return TLS_HANDSHAKE_ERROR;
258
259 case SSL_ERROR_SSL:
260 /* Protocol error - use ERR_get_error() for details.
261 * Set errno to indicate protocol-level failure. */
262 socket->tls_handshake_done = 0;
263 errno = EPROTO;
264 return TLS_HANDSHAKE_ERROR;
265
266 default:
267 /* Unknown error type - should not happen with current OpenSSL versions
268 */
269 socket->tls_handshake_done = 0;
270 errno = EIO;
271 return TLS_HANDSHAKE_ERROR;
272 }
273}

382static inline int
383tls_validate_file_path (const char *path)
384{
385 if (!path || !*path)
386 return 0;
387
388 size_t len = strlen (path);
389 if (len == 0 || len > SOCKET_TLS_MAX_PATH_LEN)
390 return 0;
391
392 /* Check for specific path traversal sequences (avoid false positives on
393 * filenames like "cert..pem") */
394 const char *traversal_patterns[]
395 = { "/../", "\\..\\", "/..\\",
396 "\\../", "/.../", "\\../", /* Added context-aware */
397 NULL };
398 for (const char **pat = traversal_patterns; *pat != NULL; ++pat)
399 {
400 if (strstr (path, *pat) != NULL)
401 return 0;
402 }
403
404 /* Reject paths starting with relative traversal */
405 if (strncmp (path, "../", 3) == 0 || strncmp (path, "..\\", 3) == 0)
406 return 0;
407
408 /* Additional symlink detection (optional, conservative: reject if detectable
409 * symlink) */
410 struct stat sb;
411 if (lstat (path, &sb) == 0)
412 {
413 if (S_ISLNK (sb.st_mode))
414 return 0; /* Reject symlinks to prevent attacks */
415 }
416 /* If lstat fails (e.g., no perm), continue validation (false negative ok for
417 * usability) */
418
419 /* Reject embedded null bytes (paranoia check).
420 Note: memchr is intentional - we're checking WITHIN the valid strlen,
421 not including the terminator. This detects strings with embedded nulls. */
422 // NOLINTNEXTLINE(bugprone-not-null-terminated-result)
423 if (memchr (path, '\0', len) != NULL)
424 return 0;
425
426 /* Check for control characters (ASCII 0-31 and 127) */
427 for (size_t i = 0; i < len; i++)
428 {
429 unsigned char c = (unsigned char)path[i];
430 if (c < 32 || c == 127)
431 return 0;
432 }
433
434 return 1;
435}

492static inline int
493tls_validate_hostname (const char *hostname)
494{
495 if (!hostname)
496 return 0;
497
498 size_t len = strlen (hostname);
499 if (len == 0 || len > SOCKET_TLS_MAX_SNI_LEN)
500 return 0;
501
502 const char *p = hostname;
503 int label_len = 0;
504 int prev_hyphen
505 = 0; /* Track if previous char was hyphen (for end-of-label check) */
506
507 while (*p)
508 {
509 if (*p == '.')
510 {
511 /* RFC 952/1123: Labels cannot be empty, exceed 63 chars, or end with
512 * hyphen */
513 if (label_len == 0 || label_len > 63 || prev_hyphen)
514 return 0;
515 label_len = 0;
516 prev_hyphen = 0;
517 }
518 else
519 {
520 if (!(isalnum ((unsigned char)*p) || *p == '-'))
521 return 0;
522 /* RFC 952/1123: Labels cannot start with hyphen */
523 if (*p == '-' && label_len == 0)
524 return 0;
525 prev_hyphen = (*p == '-');
526 label_len++;
527 if (label_len > 63)
528 return 0;
529 }
530 p++;
531 }
532
533 /* Final label: must exist, not exceed 63 chars, and not end with hyphen */
534 return (label_len > 0 && label_len <= 63 && !prev_hyphen);
535}
