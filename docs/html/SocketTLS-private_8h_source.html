306static inline TLSHandshakeState
307tls_handle_ssl_error (Socket_T socket, SSL *ssl, int ssl_result)
308{
309 int ssl_error = SSL_get_error (ssl, ssl_result);
310
311 switch (ssl_error)
312 {
313 case SSL_ERROR_NONE:
314 socket->tls_handshake_done = 1;
315 return TLS_HANDSHAKE_COMPLETE;
316
317 case SSL_ERROR_WANT_READ:
318 socket->tls_handshake_done = 0;
319 errno = EAGAIN;
320 return TLS_HANDSHAKE_WANT_READ;
321
322 case SSL_ERROR_WANT_WRITE:
323 socket->tls_handshake_done = 0;
324 errno = EAGAIN;
325 return TLS_HANDSHAKE_WANT_WRITE;
326
327 case SSL_ERROR_ZERO_RETURN:
328 /* Clean shutdown by peer - not an error per se, but connection is done
329 */
330 socket->tls_handshake_done = 0;
331 return TLS_HANDSHAKE_ERROR;
332
333 case SSL_ERROR_SYSCALL:
334 /* System call error - errno contains the actual error.
335 * If errno is 0, it typically means unexpected EOF (connection reset).
336 * Do NOT overwrite errno here - preserve it for caller diagnostics. */
337 socket->tls_handshake_done = 0;
338 if (errno == 0)
339 errno = ECONNRESET; /* Unexpected EOF treated as connection reset */
340 return TLS_HANDSHAKE_ERROR;
341
342 case SSL_ERROR_SSL:
343 /* Protocol error - use ERR_get_error() for details.
344 * Set errno to indicate protocol-level failure. */
345 socket->tls_handshake_done = 0;
346 errno = EPROTO;
347 return TLS_HANDSHAKE_ERROR;
348
349 default:
350 /* Unknown error type - should not happen with current OpenSSL versions
351 */
352 socket->tls_handshake_done = 0;
353 errno = EIO;
354 return TLS_HANDSHAKE_ERROR;
355 }
356}

469static inline int
470tls_validate_file_path (const char *path)
471{
472 if (!path || !*path)
473 return 0;
474
475 size_t len = strlen (path);
476 if (len == 0 || len > SOCKET_TLS_MAX_PATH_LEN)
477 return 0;
478
479 /* Check for specific path traversal sequences (avoid false positives on
480 * filenames like "cert..pem") */
481 const char *traversal_patterns[]
482 = { "/../", "\\..\\", "/..\\",
483 "\\../", "/.../", "\\../", /* Added context-aware */
484 NULL };
485 for (const char **pat = traversal_patterns; *pat != NULL; ++pat)
486 {
487 if (strstr (path, *pat) != NULL)
488 return 0;
489 }
490
491 /* Reject paths starting with relative traversal */
492 if (strncmp (path, "../", 3) == 0 || strncmp (path, "..\\", 3) == 0)
493 return 0;
494
495 /* Additional symlink detection (optional, conservative: reject if detectable
496 * symlink) */
497 struct stat sb;
498 if (lstat (path, &sb) == 0)
499 {
500 if (S_ISLNK (sb.st_mode))
501 return 0; /* Reject symlinks to prevent attacks */
502 }
503 /* If lstat fails (e.g., no perm), continue validation (false negative ok for
504 * usability) */
505
506 /* Reject embedded null bytes (paranoia check).
507 Note: memchr is intentional - we're checking WITHIN the valid strlen,
508 not including the terminator. This detects strings with embedded nulls. */
509 // NOLINTNEXTLINE(bugprone-not-null-terminated-result)
510 if (memchr (path, '\0', len) != NULL)
511 return 0;
512
513 /* Check for control characters (ASCII 0-31 and 127) */
514 for (size_t i = 0; i < len; i++)
515 {
516 unsigned char c = (unsigned char)path[i];
517 if (c < 32 || c == 127)
518 return 0;
519 }
520
521 return 1;
522}

580static inline int
581tls_validate_hostname (const char *hostname)
582{
583 if (!hostname)
584 return 0;
585
586 size_t len = strlen (hostname);
587 if (len == 0 || len > SOCKET_TLS_MAX_SNI_LEN)
588 return 0;
589
590 const char *p = hostname;
591 int label_len = 0;
592 int prev_hyphen
593 = 0; /* Track if previous char was hyphen (for end-of-label check) */
594
595 while (*p)
596 {
597 if (*p == '.')
598 {
599 /* RFC 952/1123: Labels cannot be empty, exceed 63 chars, or end with
600 * hyphen */
601 if (label_len == 0 || label_len > 63 || prev_hyphen)
602 return 0;
603 label_len = 0;
604 prev_hyphen = 0;
605 }
606 else
607 {
608 if (!(isalnum ((unsigned char)*p) || *p == '-'))
609 return 0;
610 /* RFC 952/1123: Labels cannot start with hyphen */
611 if (*p == '-' && label_len == 0)
612 return 0;
613 prev_hyphen = (*p == '-');
614 label_len++;
615 if (label_len > 63)
616 return 0;
617 }
618 p++;
619 }
620
621 /* Final label: must exist, not exceed 63 chars, and not end with hyphen */
622 return (label_len > 0 && label_len <= 63 && !prev_hyphen);
623}
