265static inline TLSHandshakeState
266tls_handle_ssl_error (Socket_T socket, SSL *ssl, int ssl_result)
267{
268 int ssl_error = SSL_get_error (ssl, ssl_result);
269
270 switch (ssl_error)
271 {
272 case SSL_ERROR_NONE:
273 socket->tls_handshake_done = 1;
274 return TLS_HANDSHAKE_COMPLETE;
275
276 case SSL_ERROR_WANT_READ:
277 socket->tls_handshake_done = 0;
278 errno = EAGAIN;
279 return TLS_HANDSHAKE_WANT_READ;
280
281 case SSL_ERROR_WANT_WRITE:
282 socket->tls_handshake_done = 0;
283 errno = EAGAIN;
284 return TLS_HANDSHAKE_WANT_WRITE;
285
286 case SSL_ERROR_ZERO_RETURN:
287 /* Clean shutdown by peer - not an error per se, but connection is done
288 */
289 socket->tls_handshake_done = 0;
290 return TLS_HANDSHAKE_ERROR;
291
292 case SSL_ERROR_SYSCALL:
293 /* System call error - errno contains the actual error.
294 * If errno is 0, it typically means unexpected EOF (connection reset).
295 * Do NOT overwrite errno here - preserve it for caller diagnostics. */
296 socket->tls_handshake_done = 0;
297 if (errno == 0)
298 errno = ECONNRESET; /* Unexpected EOF treated as connection reset */
299 return TLS_HANDSHAKE_ERROR;
300
301 case SSL_ERROR_SSL:
302 /* Protocol error - use ERR_get_error() for details.
303 * Set errno to indicate protocol-level failure. */
304 socket->tls_handshake_done = 0;
305 errno = EPROTO;
306 return TLS_HANDSHAKE_ERROR;
307
308 default:
309 /* Unknown error type - should not happen with current OpenSSL versions
310 */
311 socket->tls_handshake_done = 0;
312 errno = EIO;
313 return TLS_HANDSHAKE_ERROR;
314 }
315}

424static inline int
425tls_validate_file_path (const char *path)
426{
427 if (!path || !*path)
428 return 0;
429
430 size_t len = strlen (path);
431 if (len == 0 || len > SOCKET_TLS_MAX_PATH_LEN)
432 return 0;
433
434 /* Check for specific path traversal sequences (avoid false positives on
435 * filenames like "cert..pem") */
436 const char *traversal_patterns[]
437 = { "/../", "\\..\\", "/..\\",
438 "\\../", "/.../", "\\../", /* Added context-aware */
439 NULL };
440 for (const char **pat = traversal_patterns; *pat != NULL; ++pat)
441 {
442 if (strstr (path, *pat) != NULL)
443 return 0;
444 }
445
446 /* Reject paths starting with relative traversal */
447 if (strncmp (path, "../", 3) == 0 || strncmp (path, "..\\", 3) == 0)
448 return 0;
449
450 /* Additional symlink detection (optional, conservative: reject if detectable
451 * symlink) */
452 struct stat sb;
453 if (lstat (path, &sb) == 0)
454 {
455 if (S_ISLNK (sb.st_mode))
456 return 0; /* Reject symlinks to prevent attacks */
457 }
458 /* If lstat fails (e.g., no perm), continue validation (false negative ok for
459 * usability) */
460
461 /* Reject embedded null bytes (paranoia check).
462 Note: memchr is intentional - we're checking WITHIN the valid strlen,
463 not including the terminator. This detects strings with embedded nulls. */
464 // NOLINTNEXTLINE(bugprone-not-null-terminated-result)
465 if (memchr (path, '\0', len) != NULL)
466 return 0;
467
468 /* Check for control characters (ASCII 0-31 and 127) */
469 for (size_t i = 0; i < len; i++)
470 {
471 unsigned char c = (unsigned char)path[i];
472 if (c < 32 || c == 127)
473 return 0;
474 }
475
476 return 1;
477}

534static inline int
535tls_validate_hostname (const char *hostname)
536{
537 if (!hostname)
538 return 0;
539
540 size_t len = strlen (hostname);
541 if (len == 0 || len > SOCKET_TLS_MAX_SNI_LEN)
542 return 0;
543
544 const char *p = hostname;
545 int label_len = 0;
546 int prev_hyphen
547 = 0; /* Track if previous char was hyphen (for end-of-label check) */
548
549 while (*p)
550 {
551 if (*p == '.')
552 {
553 /* RFC 952/1123: Labels cannot be empty, exceed 63 chars, or end with
554 * hyphen */
555 if (label_len == 0 || label_len > 63 || prev_hyphen)
556 return 0;
557 label_len = 0;
558 prev_hyphen = 0;
559 }
560 else
561 {
562 if (!(isalnum ((unsigned char)*p) || *p == '-'))
563 return 0;
564 /* RFC 952/1123: Labels cannot start with hyphen */
565 if (*p == '-' && label_len == 0)
566 return 0;
567 prev_hyphen = (*p == '-');
568 label_len++;
569 if (label_len > 63)
570 return 0;
571 }
572 p++;
573 }
574
575 /* Final label: must exist, not exceed 63 chars, and not end with hyphen */
576 return (label_len > 0 && label_len <= 63 && !prev_hyphen);
577}
