261static inline TLSHandshakeState
262tls_handle_ssl_error (Socket_T socket, SSL *ssl, int ssl_result)
263{
264 int ssl_error = SSL_get_error (ssl, ssl_result);
265
266 switch (ssl_error)
267 {
268 case SSL_ERROR_NONE:
269 socket->tls_handshake_done = 1;
270 return TLS_HANDSHAKE_COMPLETE;
271
272 case SSL_ERROR_WANT_READ:
273 socket->tls_handshake_done = 0;
274 errno = EAGAIN;
275 return TLS_HANDSHAKE_WANT_READ;
276
277 case SSL_ERROR_WANT_WRITE:
278 socket->tls_handshake_done = 0;
279 errno = EAGAIN;
280 return TLS_HANDSHAKE_WANT_WRITE;
281
282 case SSL_ERROR_ZERO_RETURN:
283 /* Clean shutdown by peer - not an error per se, but connection is done
284 */
285 socket->tls_handshake_done = 0;
286 return TLS_HANDSHAKE_ERROR;
287
288 case SSL_ERROR_SYSCALL:
289 /* System call error - errno contains the actual error.
290 * If errno is 0, it typically means unexpected EOF (connection reset).
291 * Do NOT overwrite errno here - preserve it for caller diagnostics. */
292 socket->tls_handshake_done = 0;
293 if (errno == 0)
294 errno = ECONNRESET; /* Unexpected EOF treated as connection reset */
295 return TLS_HANDSHAKE_ERROR;
296
297 case SSL_ERROR_SSL:
298 /* Protocol error - use ERR_get_error() for details.
299 * Set errno to indicate protocol-level failure. */
300 socket->tls_handshake_done = 0;
301 errno = EPROTO;
302 return TLS_HANDSHAKE_ERROR;
303
304 default:
305 /* Unknown error type - should not happen with current OpenSSL versions
306 */
307 socket->tls_handshake_done = 0;
308 errno = EIO;
309 return TLS_HANDSHAKE_ERROR;
310 }
311}

423tls_secure_free_pkey (EVP_PKEY *pkey)
424{
425 if (!pkey)
426 return;

461static inline int
462tls_validate_hostname (const char *hostname)
463{
464 if (!hostname)
465 return 0;
466
467 size_t len = strlen (hostname);
468 if (len == 0 || len > SOCKET_TLS_MAX_SNI_LEN)
469 return 0;
470
471 const char *p = hostname;
472 int label_len = 0;
473 int prev_hyphen
474 = 0; /* Track if previous char was hyphen (for end-of-label check) */
475
476 while (*p)
477 {
478 if (*p == '.')
479 {
480 /* RFC 952/1123: Labels cannot be empty, exceed 63 chars, or end with
481 * hyphen */
482 if (label_len == 0 || label_len > 63 || prev_hyphen)
483 return 0;
484 label_len = 0;
485 prev_hyphen = 0;
486 }
487 else
488 {
489 if (!(isalnum ((unsigned char)*p) || *p == '-'))
490 return 0;
491 /* RFC 952/1123: Labels cannot start with hyphen */
492 if (*p == '-' && label_len == 0)
493 return 0;
494 prev_hyphen = (*p == '-');
495 label_len++;
496 if (label_len > 63)
497 return 0;
498 }
499 p++;
500 }
501
502 /* Final label: must exist, not exceed 63 chars, and not end with hyphen */
503 return (label_len > 0 && label_len <= 63 && !prev_hyphen);
504}
