162main (int argc, char **argv)

166 SocketHTTPServer_Config config;
167 volatile int result = 0;
168
169 /* Parse port from command line */
170 if (argc > 1)
171 {
172 port = atoi (argv[1]);
173 if (port <= 0 || port > 65535)
174 {
175 fprintf (stderr, "Invalid port: %s\n", argv[1]);
176 return 1;
177 }
178 }
179
180 /* Setup signal handlers for graceful shutdown */
181 signal (SIGINT, signal_handler);
182 signal (SIGTERM, signal_handler);
183 signal (SIGPIPE, SIG_IGN);
184
185 printf ("HTTP Server Example\n");
186 printf ("===================\n\n");
187
188 TRY
189 {
190 /* Configure server */
191 SocketHTTPServer_config_defaults (&config);
192 config.port = port;
193 config.bind_address = "0.0.0.0"; /* Listen on all interfaces */
194 config.backlog = 128;
195 config.max_connections = 100;
196 config.keepalive_timeout_ms = 60000;
197 config.request_timeout_ms = 30000;
198 config.max_header_size = 64 * 1024;
199 config.max_body_size = 10 * 1024 * 1024;
200
201 /* Create server */
202 server = SocketHTTPServer_new (&config);
203
204 /* Set request handler */
205 SocketHTTPServer_set_handler (server, request_handler, NULL);
206
207 /* Start listening */
208 if (SocketHTTPServer_start (server) < 0)
209 {
210 fprintf (stderr, "Failed to start server\n");
211 result = 1;
212 }
213 else
214 {
215 printf ("Server listening on http://0.0.0.0:%d\n", port);
216 printf ("Press Ctrl+C to stop\n\n");
217
218 /* Main event loop */
219 while (running)
220 {
221 /* Process events with 1 second timeout */
222 SocketHTTPServer_process (server, 1000);
223 }
224
225 printf ("\nShutting down...\n");
226 SocketHTTPServer_stop (server);
227 }
228 }
229 EXCEPT (SocketHTTPServer_BindFailed)
230 {
231 fprintf (stderr, "Failed to bind to port %d (in use?)\n", port);
232 result = 1;
233 }
234 EXCEPT (SocketHTTPServer_Failed)
235 {
236 fprintf (stderr, "Server error\n");
237 result = 1;
238 }
239 FINALLY
240 {
241 if (server)
242 {
243 SocketHTTPServer_free (&server);
244 }
245 }
246 END_TRY;
247
248 return result;
249}
