457static inline unsigned
458httpclient_host_hash (const char *host, int port, size_t table_size)
459{
460 size_t host_len = strlen (host);
461 unsigned host_hash
462 = socket_util_hash_djb2_ci_len (host, host_len, table_size);
463 unsigned port_hash = socket_util_hash_uint ((unsigned)port, table_size);
464 unsigned combined = host_hash ^ port_hash;
465 return socket_util_hash_uint (combined, table_size);
466}
